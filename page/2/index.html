PAT甲级1119 Pre- and Post-order Traversals(C++)

发表于 2017-03-13 | 分类于 PAT甲级 | 阅读次数

PAT甲级1119 Pre- and Post-order Traversals，原题地址：https://www.patest.cn/contests/pat-a-practise/1119
解题分析：
用中序和后序遍历得到的序列构建一个二叉树，二叉树可能并不唯一，大概的描述比较抽象，具体解释就直接写在注释里了。

#include&lt;iostream&gt;
using namespace std;

struct node{
    int key;
    struct node *left;
    struct node *right;
};

int preOrder[100] , postOrder[100];
int n;
int flag = 1;//flag标记二叉树是否唯一

// 递归构建二叉树
struct node *Construct(int index , int start , int end){
    struct node *ptr;

    // 递归出口
    if(index &gt;= n || start &gt; end){
        return NULL;
    }
    // 构建新结点
    ptr = new struct node;
    // 值为前序序列中对应的值
    ptr-&gt;key = preOrder[index];
    // 如果只有一个结点
    if(start == end){
        ptr-&gt;left = ptr-&gt;right = NULL;
        return ptr;
    }
    // 判断当前结点的后一个结点是否和后序序列的倒数第二个值相等，如果
    // 相等说明当前结点是后面所有结点的根结点，但当前结点可以是它的父
    // 结点的左孩子或右孩子
    if(postOrder[end - 1] == preOrder[index + 1]){
        // flag变为0
        flag = 0;
        // 假设是左孩子
        ptr-&gt;left = Construct(index + 1 , start , end - 1);
        ptr-&gt;right = NULL;
    }
    // 如果当前结点的后一个结点和后序序列的第一个值相等，则当前结点是它的父结点的左孩子，
    // 其他后面的结点全部位于它的父结点的右子树
    else if(postOrder[start] == preOrder[index + 1]){
        struct node *temp;
        temp = new struct node;
        temp-&gt;key = preOrder[index + 1];
        temp-&gt;left = temp-&gt;right = NULL;
        ptr-&gt;left = temp;
        // 构建右子树
        ptr-&gt;right = Construct(index + 2 , start + 1 , end - 1);
    }
    // 当前结点的后一个结点的值位于后序序列的中间
    else{
        int i;
        // 找到值的位置
        for(i = start + 1;i &lt; end - 1;i++){
            if(postOrder[i] == preOrder[index + 1])
                break;
        }
        // 构建左子树
        ptr-&gt;left = Construct(index + 1 , start , i);
        // 构建右子树
        ptr-&gt;right = Construct(index + i - start + 2 , i + 1 , end - 1);
    }
    return ptr;
}

// 中序遍历
void Inorder(struct node *root){
    if(root == NULL)
        return;
    Inorder(root-&gt;left);
    if(flag){
        cout &lt;&lt; root-&gt;key;
        flag = 0;
    }
    else{
        cout &lt;&lt; " " &lt;&lt; root-&gt;key;
    }
    Inorder(root-&gt;right);
}
int main(){
    int i;
    struct node *root;

    cin &gt;&gt; n;
    for(i = 0;i &lt; n;i++){
        cin &gt;&gt; preOrder[i]; 
    }
    for(i = 0;i &lt; n;i++){
        cin &gt;&gt; postOrder[i];
    }
    int index = 0;
    root = Construct(index , 0 , n - 1);
    if(flag){
        cout &lt;&lt; "Yes" &lt;&lt; endl;
    }
    else{
        cout &lt;&lt; "No" &lt;&lt; endl;
    }
    flag = 1;
    Inorder(root);
    cout &lt;&lt; endl;
}

解题分析：
考察并查集。

#include&lt;iostream&gt;
using namespace std;

int Find(int *arr , int n){
    if(arr[n] == -1){
        return n;
    }
    return arr[n] = Find(arr , arr[n]);
}

void Union(int *arr , int a , int b){
    arr[b] = a;
}

int main(){
    int n , k , q;
    int i , j;
    int root1 , root2;
    int unionSet[11000];
    int max = 0;
    int cnt = 0;

    for(i = 0;i &lt; 11000;i++){
        unionSet[i] = -1;
    }
    cin &gt;&gt; n;
    for(i = 0;i &lt; n;i++){
        cin &gt;&gt; k;
        int id;
        cin &gt;&gt; id;
        if(id &gt; max){
            max = id;
        }
        root1 = Find(unionSet , id);
        for(j = 1;j &lt; k;j++){
            cin &gt;&gt; id;
            if(id &gt; max){
                max = id;
            }
            root2 = Find(unionSet , id);
            if(root1 != root2){
                Union(unionSet , root1 , root2);
            }
        }
    }   
    for(i = 1;i &lt;= max;i++){
        if(unionSet[i] == -1){
            cnt++;
        }
    }
    cout &lt;&lt; cnt &lt;&lt; " " &lt;&lt; max &lt;&lt; endl;
    cin &gt;&gt; q;
    for(i = 0;i &lt; q;i++){
        int a , b;
        cin &gt;&gt; a &gt;&gt; b;
        root1 = Find(unionSet , a);
        root2 = Find(unionSet , b);
        if(root1 == root2){
            cout &lt;&lt; "Yes" &lt;&lt; endl;
        }
        else{
            cout &lt;&lt; "No" &lt;&lt; endl;
        }
    }
    return 0;
}

PAT甲级1115 Counting Nodes in a BST(C++)

发表于 2017-03-13 | 分类于 PAT甲级 | 阅读次数

PAT甲级1115 Counting Nodes in a BST，原题地址：https://www.patest.cn/contests/pat-a-practise/1115
解题分析：
构建一个BST，然后层序遍历。注意BST的定义，相同的值要放在左子树，不是右子树。。。。

#include&lt;iostream&gt;
#include&lt;queue&gt;
using namespace std;

struct node{
    int key;
    struct node *left;
    struct node *right;
};

int pre , cur;
int n;

struct node *Insert(struct node *root , int key){
    if(root == NULL){
        struct node *ptr;
        ptr = new struct node;
        ptr-&gt;key = key;
        ptr-&gt;left = ptr-&gt;right = NULL;
        return ptr;
    }
    if(root-&gt;key &gt;= key){
        root-&gt;left = Insert(root-&gt;left , key);
    }
    else{
        root-&gt;right = Insert(root-&gt;right , key);
    }
    return root;
}

void LevelOrder(struct node *root){
    queue&lt;struct node *&gt; q;

    q.push(root);
    while(1){
        cur = q.size();
        int cnt = cur;
        while(cnt){
            struct node  *t;
            t = q.front();
            q.pop();
            if(t-&gt;left != NULL){
                q.push(t-&gt;left);
            }
            if(t-&gt;right != NULL){
                q.push(t-&gt;right);
            }
            cnt--;
        }
        if(q.empty()){
            return;
        }
        else{
            pre = cur;
        }
    }
}

int main(){
    int i;
    struct node *root = NULL;

    cin &gt;&gt; n;
    for(i = 0;i &lt; n;i++){
        int num;
        cin &gt;&gt; num;
        root = Insert(root , num);
    }
    LevelOrder(root);
    cout &lt;&lt; cur &lt;&lt; " + " &lt;&lt; pre &lt;&lt; " = " &lt;&lt; cur + pre;
}

PAT甲级1114 Integer Set Partition(C++)

发表于 2017-03-13 | 分类于 PAT甲级 | 阅读次数

PAT甲级1114 Integer Set Partition，原题地址：https://www.patest.cn/contests/pat-a-practise/1113
解题分析：
DFS的简单应用。

#include&lt;iostream&gt;
#include&lt;cstdio&gt;
#include&lt;vector&gt;
#include&lt;algorithm&gt;
#define INFINITY 10000000
#define Max 11000
using namespace std;

struct record{
    int id;
    int cnt;
    double avgSets;
    double avgArea;
};

vector&lt;int&gt; G[Max];
int visited[Max];
int estate[Max];
int area[Max];
int cntFamily , cntPeople;
int totalSet , totalArea;
int minId;
int n;

void DFS(int root){
    int i;

    if(root == -1)
        return;
    visited[root] = 1;
    totalSet += estate[root];
    totalArea += area[root];
    cntPeople++;
    if(root &lt; minId)
        minId = root;
    for(i = 0;i &lt; G[root].size();i++){
        if(visited[G[root][i]] == 0){
            DFS(G[root][i]);
        }
    }
}

bool cmp(struct record a , struct record b){
    if(a.avgArea &gt; b.avgArea)
        return true;
    else if(a.avgArea &lt; b.avgArea)
        return false;
    else
        return a.id &lt; b.id;
}

int main(){
    int i;
    int id , fid , mid , k , cid;
    int s , a;
    vector&lt;struct record&gt; t;

    cin &gt;&gt; n;
    // 输入
    for(i = 0;i &lt; n;i++){
        cin &gt;&gt; id &gt;&gt; fid &gt;&gt; mid &gt;&gt; k;
        G[id].push_back(fid);
        G[id].push_back(mid);
        if(fid != -1)
            G[fid].push_back(id);
        if(mid != -1)
            G[mid].push_back(id);
        for(int j = 0;j &lt; k;j++){
            cin &gt;&gt; cid;
            G[id].push_back(cid);
            G[cid].push_back(id);
        }
        cin &gt;&gt; s &gt;&gt; a;
        estate[id] = s;
        area[id] = a;
    }
    // 遍历
    for(i = 0;i &lt; Max;i++){
        minId = INFINITY;
        totalSet = 0;
        totalArea = 0;
        cntPeople = 0;
        // 如果未被访问并且与其他点相连
        if(visited[i] == 0 &amp;&amp; !G[i].empty()){
            DFS(i);
            cntFamily++;
            struct record R;
            R.id = minId;
            R.cnt = cntPeople;
            R.avgSets = (double)totalSet / cntPeople;
            R.avgArea = (double)totalArea / cntPeople;
            t.push_back(R);
        }
    }
    sort(t.begin() , t.end() , cmp);
    // 输出
    printf("%d\n" , cntFamily);
    for(i = 0;i &lt; t.size();i++){
        printf("%04d %d %.3lf %.3lf\n" , t[i].id , t[i].cnt , t[i].avgSets , t[i].avgArea);
    }
}

PAT甲级1112 Stucked Keyboard(C++)

发表于 2017-03-13 | 分类于 PAT甲级 | 阅读次数

PAT甲级1112 Stucked Keyboard，原题地址：https://www.patest.cn/contests/pat-a-practise/1112
解题分析：
遍历字符串所有重复的连续序列，如果序列中的字符不是k的倍数，说明不是坏了的，flag标记为0，否则说明可能是坏了的，flag标记为1（暂时的），并且在容器v中搜索字符，如果没有找到则将字符放入v中。遍历结束后，flag标记仍然为1的就是坏了的，flag为0的是好的。然后依次输出v中的字符（先检查对于的flag是否为1，如果为1再输出，否则不输出）。接着输出原字符串，注意坏了的字符每k个输出一个，通过flag检查是否是坏的。

#include&lt;iostream&gt;
#include&lt;vector&gt;
using namespace std;

int flag[37];
string s;
vector&lt;char&gt; v;

// 将字符转换成对于的数组下标
int Convert(char c){
    int t;
    if(c &gt;= '0' &amp;&amp; c &lt;= '9'){
        t = c - '0';
    }
    else if(c &gt;= 'a' &amp;&amp; c &lt;= 'z'){
        t = c - 'a' + 10;
    }
    else{
        t = 36;
    }
    return t; 
}
int main(){
    int i , j;
    int k;

    cin &gt;&gt; k;
    cin &gt;&gt; s;
    for(i = 0;i &lt; 37;i++){
        flag[i] = 1;
    }
    for(i = 0;i &lt; s.length();i = j + 1){
        // 搜索重复序列
        j = i + 1;
        while(j &lt; s.length() &amp;&amp; s[j] == s[i]) j++;
        j--;
        // 重复序列长度
        int cnt = j - i + 1;
        // 转换
        int t = Convert(s[i]);
        // 好的
        if(cnt % k != 0){
            flag[t] = 0;
        }
        // 坏了，并且之前的序列也显示坏的
        else if(flag[t] == 1){
            int k;
            for(k = 0;k &lt; v.size();k++){
                if(v[k] == s[i]){
                    break;
                }
            }
            if(k == v.size()){
                v.push_back(s[i]);
            }
        }
    }
    // 输出坏了的字符
    for(i = 0;i &lt; v.size();i++){
        int t;
        t = Convert(v[i]);
        // 检查flag
        if(flag[t] == 1){
            cout &lt;&lt; v[i];
        }
    }
    cout &lt;&lt; endl;
    // 输出原字符串
    for(i = 0;i &lt; s.length();i = j + 1){
        j = i + 1;
        while(j &lt; s.length() &amp;&amp; s[j] == s[i]) j++;
        j--;
        int t = Convert(s[i]);
        if(flag[t] == 0){
            while(i &lt;= j){
                cout &lt;&lt; s[i];
                i++;
            }
        }
        else{
            t = (j - i + 1) / k;
            while(t &gt; 0){
                cout &lt;&lt; s[i];
                t--;
            }
        }
    }
    cout &lt;&lt; endl;
    return 0;
}

PAT甲级1111 Online Map(C++)

发表于 2017-03-13 | 分类于 PAT甲级 | 阅读次数

PAT甲级1111 Online Map，原题地址：https://www.patest.cn/contests/pat-a-practise/1111
解题分析：
最短路径和DFS。整张图是一个有向图，特别的就是每一条边有两个权值，一个长度，一个时间，其实没太大差别，就是代码会长一些。先用Dijkstra算出目的地的最短路径和最短时间，然后用两个DFS加剪枝就可以了。

#include&lt;iostream&gt;
#include&lt;vector&gt;
using namespace std;
#define INFINITY 10000000
#define Max 510

int L[Max][Max];
int T[Max][Max];
int Distance[Max];
int Time[Max];
int visitedDistance[Max];
int visitedTime[Max];
int fast = INFINITY , most = INFINITY;
int n , m;
int Start , End;
int currentLength , currentTime , cnt;
vector&lt;int&gt; v1 , v2;

// 计算最短路径和最短时间
void Dijkstra(){
    int i;
    int min1 , min2;
    int index1 , index2;

    while(1){
        min1 = INFINITY;
        min2 = INFINITY;
        // 选出当前路径最短的未访问点
        for(i = 0;i &lt; n;i++){
            if(Distance[i] &lt; min1 &amp;&amp; visitedDistance[i] == 0){
                min1 = Distance[i];
                index1 = i;
            }
        }
        // 选出当前时间最短的未访问点
        for(i = 0;i &lt; n;i++){
            if(Time[i] &lt; min2 &amp;&amp; visitedTime[i] == 0){
                min2 = Time[i];
                index2 = i;
            }
        }
        // 全部遍历
        if(min1 == INFINITY)
            break;
        visitedDistance[index1] = 1;
        visitedTime[index2] = 1;
        // 更新连接结点的最短路径
        for(i = 0;i &lt; n;i++){
            if(L[index1][i] &gt; 0 &amp;&amp; visitedDistance[i] == 0 &amp;&amp; L[index1][i] + Distance[index1] &lt; Distance[i]){
                Distance[i] = L[index1][i] + Distance[index1];
            }
        }
        // 更新连接结点的最短时间
        for(i = 0;i &lt; n;i++){
            if(T[index2][i] &gt; 0 &amp;&amp; visitedTime[i] == 0 &amp;&amp; T[index2][i] + Time[index2] &lt; Time[i]){
                Time[i] = T[index2][i] + Time[index2];
            }
        }
    }
}

// DFS
bool DFS1(int root){
    int i;
    bool flag = false;

    // 剪枝
    if(currentLength &gt; Distance[End]){
        return false;
    }
    // 访问到目的地结点
    if(root == End){
        if(currentTime &lt; fast){
            fast = currentTime;
            v1.clear();
            v1.push_back(root);
            return true;
        }
        else{
            return false;
        }
    }
    visitedDistance[root] = 1;
    // 依次搜索连接的结点
    for(i = 0;i &lt; n;i++){
        currentLength += L[root][i];
        currentTime += T[root][i];
        if(L[root][i] &gt; 0 &amp;&amp; visitedDistance[i] == 0){
            if(DFS1(i)){
                v1.push_back(root);
                flag = true;
            }
        }
        currentLength -= L[root][i];
        currentTime -= T[root][i];
    }
    visitedDistance[root] = 0;
    return flag;
}

bool DFS2(int root){
    int i;
    bool flag = false;

    if(currentTime &gt; Time[End]){
        return false;
    }
    if(root == End){
        if(cnt &lt; most){
            most = cnt;
            v2.clear();
            v2.push_back(root);
            return true;
        }
        else{
            return false;
        }
    }
    visitedTime[root] = 1;
    for(i = 0;i &lt; n;i++){
        currentTime += T[root][i];
        cnt++;
        if(T[root][i] &gt; 0 &amp;&amp; visitedTime[i] == 0){
            if(DFS2(i)){
                v2.push_back(root);
                flag = true;
            }
        }
        currentTime -= T[root][i];
        cnt--;
    }
    visitedTime[root] = 0;
    return flag;
}
