PAT甲级1100 Mars Numbers(C++)

发表于 2017-03-04 | 分类于 PAT甲级

PAT甲级1100 Mars Numbers，原题地址：https://www.patest.cn/contests/pat-a-practise/1100
解题分析：
进制转换。需要注意的是输入的数字是13的倍数时，输出结果应该是一位，最低位的0不需要输出，字符串转数字同理。

#include&lt;iostream&gt;
#include&lt;vector&gt;
#include&lt;string&gt;
using namespace std;

int main(){
    string arr[25] = {"tret","jan","feb","mar","apr","may","jun","jly","aug","sep","oct","nov","dec","tam","hel","maa","huh","tou","kes","hei","elo","syy","lok","mer","jou"};
    char ch;
    vector&lt;char&gt; v;
    int n , i;
    int flag;

    cin &gt;&gt; n;
    getchar();
    for(i = 0;i &lt; n;i++){
        v.clear();
        flag = 0;
        // 输入
        while((ch = getchar()) != '\n'){
            v.push_back(ch);
            if(ch == ' ')
                flag = 1;
        }
        // 如果是数字
        if(v[0] &gt;= '0' &amp;&amp; v[0] &lt;= '9'){
            int res = 0 , t = 1;
            // 计算数字大小
            for(int j = v.size() - 1;j &gt;= 0;j--){
                res += (v[j] - '0') * t;
                t *= 10;
            }
            // 如果在0到12之间
            if(res &gt;= 0 &amp;&amp; res &lt;= 12)
                cout &lt;&lt; arr[res] &lt;&lt; endl;
            // 如果是13的倍数
            else if(res % 13 == 0)
                cout &lt;&lt; arr[res / 13 + 12] &lt;&lt; endl;
            // 否则
            else
                cout&lt;&lt; arr[res / 13 + 12] &lt;&lt; " " &lt;&lt; arr[res % 13] &lt;&lt; endl;
        }
        // 如果是字符串
        else{
            string s;
            int res = 0;
            int j , k;
            // 有两个字符串（有空格）
            if(flag){
                // 第一个字符串
                for(j = 0;v[j] != ' ';j++)
                    s += v[j];
                // 计算
                for(k = 13;k &lt; 25;k++)
                    if(arr[k] == s){
                        res += (k - 12) * 13;
                        break;
                    }
                // 第二个字符串
                s = "";
                for(k = j + 1;k &lt; v.size();k++)
                    s += v[k];
                // 计算
                for(k = 0;k &lt; 13;k++)
                    if(arr[k] == s){
                        res += k;
                        break;
                    }
                // 输出
                cout &lt;&lt; res &lt;&lt; endl;
            }
            // 只有一个字符串
            else{
                // 计算
                for(j = 0;j &lt; v.size();j++)
                    s += v[j];
                // 输出
                for(j = 0;j &lt; 13;j++)
                    if(arr[j] == s){
                        cout &lt;&lt; j &lt;&lt; endl;
                        break;
                    }
                for(j = 13;j &lt; 25;j++)
                    if(arr[j] == s){
                        cout &lt;&lt; (j - 12) * 13 &lt;&lt; endl;
                        break;
                    }
            }
        }
    }
}

PAT甲级1099 Build A Binary Search Tree(C++)

发表于 2017-03-04 | 分类于 PAT甲级

PAT甲级1099 Build A Binary Search Tree，原题地址：https://www.patest.cn/contests/pat-a-practise/1099
解题分析：
给定一个序列和一个BST结构，可以构建一个唯一的BST。构建BST的关键是要计算出每一个节点它的左子树的节点个数，然后对序列排序后递归构建BST。构建好BST后层序遍历输出即可。

#include&lt;iostream&gt;
#include&lt;algorithm&gt;
#include&lt;queue&gt;
using namespace std;

struct node{
    int key;
    struct node *Left;
    struct node *Right;
};       

int arr[110];
int Left[110];
int Right[110];
int LeftNum[110];
int RightNum[110];
int n;

// 构建BST
struct node *Construct(int index , int start , int end){
    struct node *ptr;
    int t;

    if(index == -1)
        return NULL;
    t = start + LeftNum[index];
    ptr = new struct node;
    ptr-&gt;key = arr[t];
    ptr-&gt;Left = Construct(Left[index] , start , t - 1);
    ptr-&gt;Right = Construct(Right[index] , t + 1 , end);
    return ptr;
}

// 层序遍历
void LevelOrder(struct node *root){
    queue&lt;struct node&gt; q;
    struct node temp;
    int flag = 1;

    q.push(*root);
    while(!q.empty()){
        temp = q.front();
        q.pop();
        if(temp.Left != NULL)
            q.push(*temp.Left);
        if(temp.Right != NULL)
            q.push(*temp.Right);
        if(flag){
            cout &lt;&lt; temp.key;
            flag = 0;
        }
        else
            cout &lt;&lt; " " &lt;&lt; temp.key;
    }
    cout &lt;&lt; endl;
}

// 计算每一个节点的左子树的节点个数
int Calculate(int index){
    if(index == -1)
        return 0;
    LeftNum[index] = Calculate(Left[index]);
    RightNum[index] = Calculate(Right[index]);
    return LeftNum[index] + RightNum[index] + 1;
}

int main(){
    struct node *root;
    int i;

    cin &gt;&gt; n;
    for(i = 0;i &lt; n;i++){
        cin &gt;&gt; Left[i] &gt;&gt; Right[i];
    }
    for(i = 0;i &lt; n;i++){
        cin &gt;&gt; arr[i];
    }
    Calculate(0);
    sort(arr , arr + n);
    root = Construct(0 , 0 , n - 1);
    LevelOrder(root);
}

PAT甲级1098 Insertion or Heap Sort(C++)

发表于 2017-03-04 | 分类于 PAT甲级

PAT甲级1098 Insertion or Heap Sort，原题地址：https://www.patest.cn/contests/pat-a-practise/1098
解题分析：
插入排序和堆排序。难度不是很大，但堆排序比较繁琐，需要自己建堆，然后再写一个Delete函数，注意细节即可。

#include&lt;iostream&gt;
#include&lt;vector&gt;
using namespace std;

// 建堆
void ConstructHeap(vector&lt;int&gt; &amp;v){
    int i;

    for(i = v.size() / 2 - 1;i &gt;= 0;i--){
        int parent , child;
        for(parent = i;parent * 2 + 1 &lt; v.size();parent = child){
            child = parent * 2 + 1;
            if(child != v.size() - 1 &amp;&amp; v[child + 1] &gt; v[child])
                child = child + 1;
            if(v[child] &gt; v[parent]){
                int t;
                t = v[child];
                v[child] = v[parent];
                v[parent] = t;
            }
            else    
                break;
        }
    }
}

// 删除顶部元素并返回其值
int Delete(vector&lt;int&gt; &amp;v , int cnt){
    int ans;
    int t;
    int child;
    int i;

    ans = v[0];
    t = v[v.size() - cnt];
    for(i = 0;i * 2 + 1 &lt; v.size() - cnt;i = child){
        child = i * 2 + 1;
        if(child != v.size() - cnt &amp;&amp; v[child + 1] &gt; v[child])
            child = child + 1;
        if(v[child] &gt; t)
            v[i] = v[child];
        else
            break;
    }
    v[i] = t;
    return ans;
}

// 判断两个序列是否相等
bool IsEqual(vector&lt;int&gt; v1 , vector&lt;int&gt; v2){
    int i;

    for(i = 0;i &lt; v1.size();i++)
        if(v1[i] != v2[i])
            return false;
    return true;
}

// 是否是插入排序
bool IsInsertionSort(vector&lt;int&gt; &amp;v1 , vector&lt;int&gt; &amp;v2 , vector&lt;int&gt; &amp;res){
    int i , j;
    bool flag = false;

    res.clear();
    for(i = 0;i &lt; v1.size();i++)
        res.push_back(v1[i]);
    for(i = 1;i &lt; res.size();i++){
        int t = res[i];
        for(j = i - 1;j &gt;= 0;j--){
            if(res[j] &gt; t)
                res[j + 1] = res[j];
            else
                break;
        }
        res[j + 1] = t;
        if(flag)
            return true;
        if(IsEqual(v2 , res))
            flag = true;
    }
    return false;
}

// 是否是堆排序
bool IsHeapSort(vector&lt;int&gt; &amp;v1 , vector&lt;int&gt; &amp;v2 , vector&lt;int&gt; &amp;res){
    int i;
    bool flag = false;

    res.clear();
    for(i = 0;i &lt; v1.size();i++)
        res.push_back(v1[i]);
    ConstructHeap(res);
    for(i = 1;i &lt;= res.size();i++){
        int t = Delete(res , i);
        res[res.size() - i] = t;
        if(flag)
            return true;
        if(IsEqual(v2 , res))
            flag = true;
    }
    return false;
}

int main(){
    vector&lt;int&gt; v1;
    vector&lt;int&gt; v2;
    vector&lt;int&gt; res;
    int n;
    int i;
    int num;

    cin &gt;&gt; n;
    for(i = 0;i &lt; n;i++){
        cin &gt;&gt; num;
        v1.push_back(num);
    }
    for(i = 0;i &lt; n;i++){
        cin &gt;&gt; num;
        v2.push_back(num);
    }
    // 判断是否是插入排序
    if(IsInsertionSort(v1 , v2 , res)){
        cout &lt;&lt; "Insertion Sort" &lt;&lt; endl;
        for(i = 0;i &lt; n;i++){
            if(i != n - 1)
                cout &lt;&lt; res[i] &lt;&lt; " ";
            else
                cout &lt;&lt; res[i] &lt;&lt; endl;
        }
    }
    // 判断是否是堆排序
    else if(IsHeapSort(v1 , v2 , res)){
        cout &lt;&lt; "Heap Sort" &lt;&lt; endl;
        for(i = 0;i &lt; n;i++){
            if(i != n - 1)
                cout &lt;&lt; res[i] &lt;&lt; " ";
            else
                cout &lt;&lt; res[i] &lt;&lt; endl;
        }
    }
}

PAT甲级1097 Deduplication on a Linked List(C++)

发表于 2017-03-04 | 分类于 PAT甲级

PAT甲级1097 Deduplication on a Linked List，原题地址：https://www.patest.cn/contests/pat-a-practise/1097
解题分析：
链表题。从头节点开始遍历整个链表，遇到一个节点，先在check里查询之前是否有相同绝对值的节点出现，如果没有则将该节点加入到第一个结果数组里，并且将对应的check数组的值加一表示该绝对值出现过了，如果有则将该节点加入到第二个结果数组。最后依次输出即可。

#include&lt;iostream&gt;
#include&lt;cstdio&gt;
#include&lt;vector&gt;
using namespace std;

struct node{
    long address;
    long key;
    long next;
};

struct node arr[110000];
vector&lt;struct node&gt; res1;
vector&lt;struct node&gt; res2;
long check[11000];
long head , n;

int main(){
    long i;

    cin &gt;&gt; head &gt;&gt; n;
    for(i = 0;i &lt; n;i++){
        long address , key , next;
        cin &gt;&gt; address &gt;&gt; key &gt;&gt; next;
        arr[address].address = address;
        arr[address].key = key;
        arr[address].next = next;
    }
    // 遍历链表
    for(i = head;i != -1;i = arr[i].next){
        long t = arr[i].key &gt; 0 ? arr[i].key : -arr[i].key;
        // 如果没有出翔
        if(check[t] == 0){
            check[t]++;
            res1.push_back(arr[i]);
        }
        // 如果出现过
        else{
            res2.push_back(arr[i]);
        }
    }
    // 输出
    for(i = 0;i &lt; res1.size();i++){
        printf("%05ld %ld " , res1[i].address , res1[i].key);        
        if(i != res1.size() - 1)
            printf("%05ld\n" , res1[i + 1].address);                
        else
            printf("-1\n");
    }
    for(i = 0;i &lt; res2.size();i++){
        printf("%05ld %ld " , res2[i].address , res2[i].key);        
        if(i != res2.size() - 1)
            printf("%05ld\n" , res2[i + 1].address);                
        else    
            printf("-1\n");
    }
}

PAT甲级1096 Consecutive Factors(C++)

发表于 2017-03-02 | 分类于 PAT甲级

PAT甲级1096 Consecutive Factors，原题地址：https://www.patest.cn/contests/pat-a-practise/1096
解题分析：
暴力搜索，每一种情况都计算过去，取最大的就行了。为了不超时，还要将n开一个根号。

#include&lt;iostream&gt;
#include&lt;cmath&gt;
using namespace std;

int main(){
    long n;
    long i;
    long temp;
    long index;
    long len = 0;
    long max = 0;

    cin &gt;&gt; n;
    // n需要开根号
    for(i = 2;i &lt; sqrt(n);i++){
        len = 0;
        temp = i;
        // 如果n能被i整除
        if(n % i == 0){
            len++;
            // 尝试下一个连续值
            for(int j = i + 1;j &lt; n;j++){
                temp *= j;
                // 如果能整除n，长度加一
                if(n % temp == 0)
                    len++;
                // 否则
                else
                    break;
            }
            // 如果长度比max大
            if(len &gt; max){
                max = len;
                index = i;
            }
        }
    }
    // 2和3
    if(max == 0){
        max = 1;
        index = n;
    }
    // 输出
    cout &lt;&lt; max &lt;&lt; endl;
    for(i = 0;i &lt; max;i++){
        if(i &gt; 0)
            cout &lt;&lt; "*" &lt;&lt; index + i;
        else
            cout &lt;&lt; index + i;
    }
    cout &lt;&lt; endl;
}

PAT甲级1095 Cars on Campus(C++)

发表于 2017-03-02 | 分类于 PAT甲级

PAT甲级1095 Cars on Campus，原题地址：https://www.patest.cn/contests/pat-a-practise/1095
解题分析：
排序题。先把所有数据排序，首先按车牌号排序，如果车牌号相同再按时间排序，排好后删去没有配对的记录，得到一个新纪录，这个过程中我们可以得到停留时间最长的车和最长停留时间。然后将新得到的记录表按时间排序，每次查询都在上次查询的结束位置开始，有车进来num就加一，有车出去就减一，最后输出即可。

#include&lt;iostream&gt;
#include&lt;cstdio&gt;
#include&lt;vector&gt;
#include&lt;string&gt;
#include&lt;algorithm&gt;
using namespace std;

struct record{
    string plate;
    int t;
    int status;
};

vector&lt;struct record&gt; v1;
vector&lt;struct record&gt; v2;
vector&lt;string&gt; l;
int pre , num;
int Max;
long n , k;

// 配对
void Update(){
    int i;
    int sum = 0;

    for(i = 0;i &lt; n;i++){
        // 如果下一辆车的车牌号跟当前的不同，说明车牌号相同的车已经全部找到了
        if(i == n - 1 || v1[i + 1].plate != v1[i].plate){
            // 将该车牌号的车的停留时间和最长停留时间比较
            if(sum &gt; Max){
                Max = sum;
                l.clear();
                l.push_back(v1[i].plate);
            }
            else if(sum == Max){
                l.push_back(v1[i].plate);
            }
            sum = 0;
        }
        // 如果下一辆车的车牌号跟当前的相同，并且当前的车为进入，下一辆为出去，说明可配对。
        else if(v1[i].status == 1 &amp;&amp; v1[i + 1].status == 0){
            // 将可配对的记录加入到新纪录中
            v2.push_back(v1[i]);
            v2.push_back(v1[i + 1]);
            // 计算停留时间
            sum += v1[i + 1].t - v1[i].t;
        }
    }
}

// 查询
void Query(int t){
    int i;

    // 从上一次结束的位置开始查询
    for(i = pre;i &lt; v2.size() &amp;&amp; v2[i].t &lt;= t;i++){
        // 进入
        if(v2[i].status == 1)
            num++;
        // 出去
        else
            num--;
    }
    pre = i;
    printf("%d\n" , num);
}

bool cmp1(struct record a , struct record b){
    if(a.plate != b.plate)
        return a.plate &lt; b.plate;
    else
        return a.t &lt; b.t;
}

bool cmp2(struct record a , struct record b){
    return a.t &lt; b.t;
}

int main(){
    int hour , minute , second;
    int t;
    long i;
    string plate;
    string status;
    struct record R;

    scanf("%ld%ld" , &amp;n , &amp;k);
    // 输入
    for(i = 0;i &lt; n;i++){
        cin &gt;&gt; plate;
        scanf("%d:%d:%d" , &amp;hour , &amp;minute , &amp;second);
        cin &gt;&gt; status;

        R.plate = plate;
        R.t = hour * 3600 + minute * 60 + second;
        R.status = status[0] == 'i' ? 1 : 0;
        v1.push_back(R);
    }
    // 排序
    sort(v1.begin() , v1.end() , cmp1);
    // 更新记录，删去不配对记录
    Update();
    // 排序
    sort(v2.begin() , v2.end() , cmp2);
    // 查询
    for(i = 0;i &lt; k;i++){
        scanf("%d:%d:%d" , &amp;hour , &amp;minute , &amp;second);
        t = hour * 3600 + minute * 60 + second;
        Query(t);
    }
    // 输出
    for(i = 0;i &lt; l.size();i++)
        cout &lt;&lt; l[i] &lt;&lt; " ";
    printf("%02d:%02d:%02d\n" , Max / 3600 , Max % 3600 / 60 , Max % 3600 % 60);
}

PAT甲级1094 The Largest Generation(C++)

发表于 2017-03-02 | 分类于 PAT甲级

PAT甲级1094 The Largest Generation，原题地址：https://www.patest.cn/contests/pat-a-practise/1094
解题分析：
简单的BFS，对于队列中的元素，我们要一层一层处理，不能一个一个处理，其他就不解释了。

#include&lt;iostream&gt;
#include&lt;queue&gt;
using namespace std;
#define Max 110

int G[Max][Max];
int visited[Max];
int n , m;
int cnt , lev;

void BFS(int root){
    queue&lt;int&gt; q;
    int ccnt , clev = 1;
    int index;

    q.push(root);
    visited[root] = 1;
    while(!q.empty()){
        ccnt = q.size();
        if(ccnt &gt; cnt){
            cnt = ccnt;
            lev = clev;
        }
        while(ccnt &gt; 0){
            index = q.front();
            q.pop();
            for(int i = 1;i &lt;= n;i++){
                if(G[index][i] == 1 &amp;&amp; visited[i] == 0){
                    q.push(i);
                    visited[i] = 1;
                }
            } 
            ccnt--;
        }
        clev++;
    }
}

int main(){
    int i , j , k;
    int pid , cid;

     cin &gt;&gt; n &gt;&gt; m;
     for(i = 0;i &lt; m;i++){
         cin &gt;&gt; pid &gt;&gt; k;
         for(j = 0;j &lt; k;j++){
             cin &gt;&gt; cid;
             G[pid][cid] = 1;
         }
     }
     BFS(1);
     cout &lt;&lt; cnt &lt;&lt; " " &lt;&lt; lev &lt;&lt; endl;
}

PAT甲级1093 Count PAT's(C++)

发表于 2017-03-01 | 分类于 PAT甲级

PAT甲级1093 Count PAT’s，原题地址：https://www.patest.cn/contests/pat-a-practise/1093
解题分析：
动态规划问题。假设我们有一个编号为0到n的序列，长度为n+1，我们从最后一个元素开始考虑，如果这个元素是T，那么PAT[n + 1]（表示长度为n+1的序列的PAT的个数）等于PAT[n]加上PA[n]（表示长度为n的序列的PA的个数）。如果这个元素是A，那么PA[n + 1]等于PA[n]加上P[n]（表示长度为n的序列的P的个数）。如果这个元素是P，那么P[n + 1]等于P[n] + 1。有了这些关系，我们就可以自底向上求解出长度为n+1时PAT的个数。另外，可以在输入的时候直接判断出最后一个T的位置，这样能减少一些不必要的运算。

#include&lt;iostream&gt;
#define Max 110000
using namespace std;
typedef long long ll;

char s[Max];
ll P[Max];
ll PA[Max];
ll PAT[Max];
ll n;

int main(){
    ll i;
    char c;

    i = 0;
    while((c = getchar()) != '\n'){
        s[i] = c;
        // 得到最后一个T的位置
        if(c == 'T')
            n = i;
        i++;
    }
    // 自底向上求解
    for(i = 1;i &lt;= n + 1;i++){
        // 元素为P
        if(s[i - 1] == 'P'){
            P[i] = P[i - 1] + 1;
            PA[i] = PA[i - 1];
            PAT[i] = PAT[i - 1];
        }
        // 元素为A
        if(s[i - 1] == 'A'){
            PA[i] = PA[i - 1] + P[i - 1];
            P[i] = P[i - 1];
            PAT[i] = PAT[i - 1];
        }
        // 元素为T
        if(s[i - 1] == 'T'){
            PAT[i] = PAT[i - 1] + PA[i - 1];
            P[i] = P[i - 1];
            PA[i] = PA[i - 1];
        }
    }
    cout &lt;&lt; PAT[n + 1] % 1000000007;
}

PAT甲级1092 To Buy or Not to Buy(C++)

发表于 2017-03-01 | 分类于 PAT甲级

PAT甲级1092 To Buy or Not to Buy，原题地址：https://www.patest.cn/contests/pat-a-practise/1092
解题分析：
简单的一道题，输入第一个字符串的时候将每个字符转换成数组的下标，其对应的值加一。输入第二个的时候，同样把字符转成
下标，对应的值减一，如果对应的值是0，说明不满足需求，flag设为0，并记录还需要多少珠子。最后输出即可。

#include&lt;iostream&gt;
using namespace std;

int cnt[100];
int total , need;

int main(){
    char c;
    int index;
    int flag = 1;

    while((c = getchar()) != '\n'){
        if(c &gt;= '0' &amp;&amp; c &lt;= '9')
            cnt[c - '0']++;
        else if(c &gt;= 'a' &amp;&amp; c &lt;= 'z')
            cnt[c - 'a' + 10]++;
        else
            cnt[c - 'A' + 36]++;
        total++;
    }
    while((c = getchar()) != '\n'){
        if(c &gt;= '0' &amp;&amp; c &lt;= '9')
            index = c - '0';
        else if(c &gt;= 'a' &amp;&amp; c &lt;= 'z')
            index = c - 'a' + 10;
        else
            index = c - 'A' + 36;
        if(cnt[index] &gt; 0){
            cnt[index]--;
            total--;
        }
        else{
            flag = 0;
            need++;
        }
    }
    if(flag)
        cout &lt;&lt; "Yes " &lt;&lt; total &lt;&lt; endl;
    else
        cout &lt;&lt; "No " &lt;&lt; need &lt;&lt; endl; 
}

PAT甲级1091 Acute Stroke(C++)

发表于 2017-03-01 | 分类于 PAT甲级

PAT甲级1091 Acute Stroke，原题地址：https://www.patest.cn/contests/pat-a-practise/1091
解题分析：
题目意思有些难以理解，大致是说大脑里有很多个片段，每个片段有很多个像素点，这些像素点构成的是一个m*n的矩阵，片段是竖直叠加起来的，因此在空间上是三维结构。题目中的图片的意思是每一个像素点，它的上下左右前后的点和它相连，以此类推可以得到一片区域，我们要做的就是计算区域内的1的个数，如果比下限大，就加到总数里面。为了实现这个目的，我们需要用BFS对一个点进行搜索，并在搜索的时候标记已经搜索过的点，避免重复搜索。遍历所有的像素点，使用BFS就可以得到总数了。

#include&lt;cstdio&gt;
#include&lt;queue&gt;
using namespace std;

struct record{
    long i;
    long j;
    long k;
};
int G[1300][130][70];
int visited[1300][130][70];
queue&lt;struct record&gt; q;
long cnt;
long m , n , l , t;

void BFS(long i , long j , long k){
    long temp = 0;
    long a , b , c;
    struct record R;

    R.i = i;
    R.j = j;
    R.k = k;
    q.push(R);
    visited[i][j][k] = 1;
    // 队列不为空
    while(!q.empty()){
        R = q.front();
        q.pop();
        temp++;
        a = R.i;
        b = R.j;
        c = R.k;
        // 后
        if(a - 1 &gt;= 0 &amp;&amp; G[a - 1][b][c] == 1 &amp;&amp; visited[a - 1][b][c] == 0){
            R.i = a - 1;
            R.j = b;
            R.k = c;
            q.push(R);
            visited[R.i][R.j][R.k] = 1;
        }
        // 前
        if(a + 1 &lt; m &amp;&amp; G[a + 1][b][c] == 1 &amp;&amp; visited[a + 1][b][c] == 0){
            R.i = a + 1;
            R.j = b;
            R.k = c;
            q.push(R);
            visited[R.i][R.j][R.k] = 1;
        }
        // 左
        if(b - 1 &gt;= 0 &amp;&amp; G[a][b - 1][c] == 1 &amp;&amp; visited[a][b - 1][c] == 0){
            R.i = a;
            R.j = b - 1;
            R.k = c;
            q.push(R);
            visited[R.i][R.j][R.k] = 1;
        }
        // 右
        if(b + 1 &lt; n &amp;&amp; G[a][b + 1][c] == 1 &amp;&amp; visited[a][b + 1][c] == 0){
            R.i = a;
            R.j = b + 1;
            R.k = c;
            q.push(R);
            visited[R.i][R.j][R.k] = 1;
        }
        // 下
        if(c - 1 &gt;= 0 &amp;&amp; G[a][b][c - 1] == 1 &amp;&amp; visited[a][b][c - 1] == 0){
            R.i = a;
            R.j = b;
            R.k = c - 1;
            q.push(R);
            visited[R.i][R.j][R.k] = 1;
        }
        // 上
        if(c + 1 &lt; l &amp;&amp; G[a][b][c + 1] == 1 &amp;&amp; visited[a][b][c + 1] == 0){
            R.i = a;
            R.j = b;
            R.k = c + 1;
            q.push(R);
            visited[R.i][R.j][R.k] = 1;
        }
    }
    if(temp &gt;= t)
        cnt += temp;
}
int main(){
    long i , j , k;

    scanf("%ld%ld%ld%ld" , &amp;m , &amp;n , &amp;l , &amp;t);
    // 输入
    for(k = 0;k &lt; l;k++){
        for(i = 0;i &lt; m;i++)
            for(j = 0;j &lt; n;j++)
                scanf("%d" , &amp;G[i][j][k]);
    }    
    // 遍历搜索点
    for(k = 0;k &lt; l;k++){
        for(i = 0;i &lt; m;i++)
            for(j = 0;j &lt; n;j++)
                if(visited[i][j][k] == 0 &amp;&amp; G[i][j][k] == 1)
                    BFS(i , j , k);
    }
    printf("%ld\n" , cnt);
}
