PAT甲级1082 Read Number in Chinese(C++)

发表于 2017-02-26 | 分类于 PAT甲级

PAT甲级1082 Read Number in Chinese，原题地址：https://www.patest.cn/contests/pat-a-practise/1082
解题分析：
分长度讨论，0的情况特殊处理，输出的时候注意空格就好了。

#include&lt;iostream&gt;
#include&lt;string&gt;
using namespace std;

int main(){
    string s;
    string num[10] = {"ling" , "yi" , "er" , "san" , "si" , "wu" , "liu" , "qi" , "ba" , "jiu"};
    string loc[10] = {" " , " " , "Shi" , "Bai" , "Qian" , "Wan" , "Shi" , "Bai" , "Qian" , "Yi"};
    int len;
    int i = 0;
    int flag;
    int temp;

    cin &gt;&gt; s;
    len = s.length();
    // 负数
    if(s[0] == '-'){
        cout &lt;&lt; "Fu";
        len--;
    }
    while(s[i] == '-' || s[i] == '0')
        i++;
    temp = i;
    // 遍历各个位的数
    for(;i &lt; s.length();i++){
        // 0的情况
        if(s[i] == '0'){
            flag = 0;
            int j;
            // 寻找之后有没有非0的数
            for(j = i + 1;j &lt; s.length();j++){
                if(s[j] != '0'){
                    flag = 1;
                    break;   
                }
            }
            // 如果之后有非0的数
            if(flag){
                cout &lt;&lt; " ling";
                len -= j - i;
                i = j - 1;
            }
            // 如果没有非0的数
            else{
                cout &lt;&lt; endl;
                return 0;
            }
        }
        // 非0的情况
        else{
            // 注意空格的输出
            if(i != temp || s[0] == '-')
                cout &lt;&lt; " ";
            // 输出数字
            cout &lt;&lt; num[s[i] - '0'];
            // 长度为9
            if(len == 9){
                cout &lt;&lt; " Yi";
            }
            // 长度在5和8之间
            if(len &gt; 5 &amp;&amp; len &lt;= 8){
                cout &lt;&lt; " ";
                flag = 0;
                int j;
                for(j = i + 1;j &lt; s.length() - 4;j++){
                    if(s[j] != '0')
                        flag = 1;
                }
                if(flag)
                    cout &lt;&lt; loc[len];
                else
                    cout &lt;&lt; loc[len] &lt;&lt; " " &lt;&lt; "Wan";
            }
            // 长度在1和5之间
            if(len &gt; 1 &amp;&amp; len &lt;= 5){
                cout &lt;&lt; " ";
                cout &lt;&lt; loc[len];
            }
            // 长度为1
            if(len == 1){
                cout &lt;&lt; endl;
                return 0;
            }
            len--;
        }
    }
    // 输入为0
    cout &lt;&lt; "ling" &lt;&lt; endl;
    return 0;
}

PAT甲级1081 Rational Sum(C++)

发表于 2017-02-26 | 分类于 PAT甲级

PAT甲级1081 Rational Sum，原题地址：https://www.patest.cn/contests/pat-a-practise/1081
解题分析：
计算分数的和，一个一个加就可以了，计算两个分数的和时先计算分母的最小公倍数，再计算出结果。输出前先把分子分母简化，再输出。

#include&lt;cstdio&gt;

// 计算最小公倍数
long GetCommon(long a , long b){
    if(a &gt;= b &amp;&amp; a % b == 0)
        return a;
    else if(a &lt; b &amp;&amp; b % a == 0)
        return b;
    else
        return a * b;
}

// 简化分子分母
void DeleteCommon(long &amp;a , long &amp;b){
    int i;

    for(i = 2;i &lt;= a &amp;&amp; i &lt;= b;){
        if(a % i == 0 &amp;&amp; b % i == 0){
            a /= i;
            b /= i;
        }
        else
            i++;
    }
}
int main(){
    long n[100] , d[100];
    long nres = 0 , dres = 1;
    long temp;
    int m;
    int i;

    scanf("%d" , &amp;m);
    for(i = 0;i &lt; m;i++){
        scanf("%ld/%ld" , &amp;n[i] , &amp;d[i]);
        // 计算两个分数的和
        temp = GetCommon(dres , d[i]);
        nres *= temp / dres;
        n[i] *= temp / d[i];
        dres = temp;
        nres = nres + n[i];
    }
    DeleteCommon(nres , dres);
    // 输出
    if(nres % dres == 0)
        printf("%ld\n" , nres / dres);
    else if(nres / dres == 0)
        printf("%ld/%ld\n" , nres , dres);
    else
        printf("%ld %ld/%ld\n" , nres / dres , nres % dres , dres);
}

关于“数据结构”概念的再理解

发表于 2017-02-26 | 分类于 数据结构

转载请加上链接http://www.huangyusheng.com/关于数据结构概念的再理解/

引言

之前写过一篇文章叫做“关于数据结构概念的理解”，其中简单的记录了我对数据结构这一概念的一些见解。但在后来我又查阅了一些资料，思考之后发现之前的文章中并没有清晰的阐明什么是数据结构，因此又有了这篇文章。下面是我对数据结构概念的新的理解。

关于数据结构，它的名词定义是

数据结构是指相互之间存在着一种或多种关系的数据元素的集合和该集合中数据元素的关系组成。记为:
Data_Structure=(D,R)，其中D指数据元素的集合，R指数据元素关系的有限集合。

这个定义看起来有些不知所然，所以我更喜欢Robert L.Kruse在《数据结构与程序设计》一书给出的解释：

数据结构的设计分为抽象层、数据结构层和实现层。其中，抽象层是指抽象数据类型层，主要讨论数据的逻辑结构和运算，数据结构层和实现层讨论数据结构的表示和在计算机内存储的细节以及运算的实现。

阐释

1、抽象层

抽象层其实就是抽象数据类型层。抽象数据类型，也就是所谓的ADT(Abstract Data Type)，是指一个数学模型和定义在该模型上的一组操作。记得在刚学数据结构的时候，老师上课给出的第一个概念就是ADT，当时由于并不了解数据结构，只沉浸在学习堆栈、队列、树这些听起来很牛逼的东西上，对ADT也就是一听而过，现在才想起来ADT是多么重要！

一般来说，计算机在解决问题时，大致需要这么几个步骤：从具体问题中抽象出一个适当数学模型，然后根据模型设计适当的算法，最后编程、调试直到解决问题。从具体问题中抽象出来的数学模型代表了问题中所有数据元素之间的相互关系，这些关系就构成了一种逻辑结构。

逻辑结构分为四种：
1.集合
各元素之间除了同处于一个集合的相互关系外，没有其他的关系
2.线性结构
元素之间存在一对一的相互关系
3.树状结构
元素之间存在一对多的相互关系
4.图形结构
元素之间存在多对多的相互关系

至于为什么就是这么四种结构，我觉得是跟现实生活中人们遇到的问题有关，大部分问题抽象出来的模型都是四种结构中的一种或多种。

现在，我们有了数据的逻辑结构，剩下的就是在这些逻辑结构上的操作了。没有操作的逻辑结构没有任何意义，你只能看看不能对它有任何的想法，那还有啥意思嘞。

不同的逻辑结构可以有很多不同的操作，但必须包含以下的基本操作：
1.查找数据元素
2.插入数据元素
3.删除数据元素

对于同一个逻辑结构，不同的操作可以使它称为不同的数据结构。用二叉树来举个例子，如果插入元素时元素的值比根节点的值小就插在左边，比它大就插在右边，这样一种插入操作就形成了二叉搜索树这种数据结构，如果插入时还要保持树的平衡，比如加上旋转操作，就构成了AVL树这种数据结构。有些数据结构还隐含了某种规则约束，规定用户使用给定的操作来对数据结构进行操作，比如堆栈、队列。

ADT就是数据的逻辑结构和定义在其上的操作的集合，之所以给它一个“抽象”的前缀，是因为ADT并不涉及到任何实现，它并不关心数据的类型，逻辑结构如何在计算中表示和储存，也并不关心这些操作该如何实现，它只是一种抽象。因此，我觉得C++中的STL就是多种ADT的集合，我们使用STL的时候不需要关心我们需要的数据结构是如何实现的，也不需要知道这些操作是怎么实现的，我们只要给定数据类型，知道使用它们就能完成任务就可以了。

2、数据结构层

数据结构层对应数据结构中的储存结构，它实现了逻辑结构在计算机中的表示和储存。

数据结构的储存结构也分为四种：
1.顺序储存
逻辑上相邻的点被储存在物理上相邻的位置上，通常用数组来实现。
2.链接储存
逻辑上相邻的点在物理位置并不相邻，通过指针可以访问相邻的点。
3.索引储存
建立一个索引表，通过关键字在索引表中索引，找到对应值的储存地址。
4.散列储存
直接对关键字进行计算得出对于值的储存地址。

不同的逻辑结构有不同的储存结构，同一种逻辑结构也可以有不同的储存方式。比如，堆的逻辑结构是二叉树，但能通过线性化使用顺序储存，而不是链接储存。不同的储存结构有着不同的运算效率，根据情况选择合适的储存结构。

3、实现层

在实现层中，我们需要通过程序设计语言实现在抽象层中定义的运算，它是具体的。运算的实现依赖于程序设计语言以及数据结构的储存结构，它的效率和数据结构的逻辑结构和储存结构有着很大的关联。

结语

讲到这里，数据结构这一概念应该算是讲清楚了，数据结构就是值数据的逻辑结构、储存结构和运算。数据的逻辑结构和抽象的运算对于数据结构设计的抽象层(ADT)，储存结构对于数据结构层，运算的实现对应实现层。我们所学习的都是计算机科学家幸幸苦苦研究出来的成果，它们都具有很广泛的应用，其实只要你足够牛，自己定义一个数据结构也未尝不可。

还有一个问题是，我们为什么要学习数据结构呢？

鄙人知识量很少，也没开发过什么项目，但单从编程的角度看，遇到一个问题的时候，如果不知道一些数据结构，面对这些问题完全就是束手无策，就算最后能解决，那代码一定惨不忍睹。因此，我认为数据结构是我们解决问题所必需掌握的，只有知道一些数据结构的知识，在遇到问题的时候就可以将问题很好的抽象出来，然后将数据组成一些结构，有了数据结构之后，自然而然地可以想出算法，然后我们再编程实现它们就可以解决问题啦。

以上。

PAT甲级1080 Graduate Admission(C++)

PAT甲级1080 Graduate Admission，原题地址：https://www.patest.cn/contests/pat-a-practise/1080
解题分析：
这题不想解释了，排序模拟一下即可。

#include&lt;iostream&gt;
#include&lt;vector&gt;
#include&lt;algorithm&gt;
using namespace std;

int g[40000][2] , c[40000][5] , cnt[100];
long ans[40000];
vector&lt;long&gt; s[100];
long n , m , k;

bool cmp(long a , long b){
    return g[a][1] == g[b][1] ? g[a][0] &gt; g[b][0] : g[a][1] &gt; g[b][1];
}

int main(){
    long i , j;

    cin &gt;&gt; n &gt;&gt; m &gt;&gt; k;
    for(i = 0;i &lt; m;i++)
        cin &gt;&gt; cnt[i];
    for(i = 0;i &lt; n;i++){
        cin &gt;&gt; g[i][0] &gt;&gt; g[i][1];
        g[i][1] += g[i][0];
        for(j = 0;j &lt; k;j++)
            cin &gt;&gt; c[i][j];
        ans[i] = i;
    }
    // 排序
    sort(ans , ans + n , cmp);
    // 一个一个处理
    for(i = 0;i &lt; n;i++){
        // 处理每个志愿
        for(j = 0;j &lt; k;j++){
            int x = c[ans[i]][j];
            // 有空位
            if(s[x].size() &lt; cnt[x]){
                s[x].push_back(ans[i]);
                break;
            }
            else{
                int temp = s[x][s[x].size() - 1];
                // 排名相等
                if(g[temp][1] == g[ans[i]][1] &amp;&amp; g[temp][0] == g[ans[i]][0]){
                    s[x].push_back(ans[i]);
                    break;
                }
            }
        }
    }
    // 输出
    for(i = 0;i &lt; m;i++){
        if(s[i].size() == 0)
            cout &lt;&lt; endl;
        else{
            // 排序
            sort(s[i].begin() , s[i].end());
            for(j = 0;j &lt; s[i].size();j++){
                if(j != 0)
                    cout &lt;&lt; " ";
                cout &lt;&lt; s[i][j];
            }
            cout &lt;&lt; endl;
        }
    }
}

PAT甲级1079 Total Sales of Supply Chain(C++)

发表于 2017-02-25 | 分类于 PAT甲级

PAT甲级1079 Total Sales of Supply Chain，原题地址：https://www.patest.cn/contests/pat-a-practise/1079
解题分析：
DFS。从根节点出发，深度优先搜索整张图，就可以计算出每个节点卖出商品时比原价高多少百分比。然后将每个零售商的商品数量乘以单价再乘上对应的溢价百分比，就可以得到最终的销售额了。

#include&lt;iostream&gt;
#include&lt;vector&gt;
#include&lt;cstdio&gt;
using namespace std;
#define Max 110000

struct record{
    long id;
    double amount;
};

vector&lt; vector&lt;long&gt; &gt; G;
vector&lt;struct record&gt; retailers;
long n;
double p , r;
double rate[Max];

// 深度优先搜索
void DFS(long n , double currentRate){
    double temp;

    // 供应商卖出商品时就是以原价卖出，不加价
    if(n == 0)
        temp = 1;
    // 其他商户都要加价
    else
        temp = currentRate * (1 + r / 100);
    // 储存溢价比
    rate[n] = temp;
    // 如果是零售商则直接返回
    if(G[n].size() == 0)
        return;
    // 如果是经销商，则继续搜索
    for(vector&lt;long&gt;::iterator it = G[n].begin();it != G[n].end();it++)
        DFS(*it , temp);
}
int main(){
    long i , j;
    long k , id;
    double amount;
    double res = 0;
    vector&lt;long&gt; temp;
    struct record R;

    cin &gt;&gt; n &gt;&gt; p &gt;&gt; r;
    for(i = 0;i &lt; n;i++){
        cin &gt;&gt; k;
        // 储存零售商信息
        if(k == 0){
            cin &gt;&gt; amount;
            R.id = i;
            R.amount = amount;
            retailers.push_back(R);
        }
        // 储存整个销售网络
        for(j = 0;j &lt; k;j++){
            cin &gt;&gt; id;
            temp.push_back(id);
        }
        G.push_back(temp);
        temp.clear();
    }
    // 搜索
    DFS(0 , 1);
    // 计算输出
    for(vector&lt;struct record&gt;::iterator it = retailers.begin();it != retailers.end();it++)
        res += (*it).amount * p * rate[(*it).id];
    printf("%.1f\n" , res);
}

PAT甲级1078 Hashing(C++)

发表于 2017-02-25 | 分类于 PAT甲级

PAT甲级1078 Hashing，原题地址：https://www.patest.cn/contests/pat-a-practise/1078
解题分析：
哈希表的平方探测法。首先简单处理一下素数，更新表的大小。然后对每个输入的值，用平方探测找到对应的位置，返回位置的索引，如果没找到返回－1。平方探测最多探测m次，m次之内没探测到就是没有位置。

#include&lt;iostream&gt;
#include&lt;vector&gt;
using namespace std;
#define Max 11000

int hashTable[Max];

// 判断素数
bool IsPrime(int n){
    int i;
    if(n == 0 || n == 1)
        return false;
    for(i = 2;i &lt; n;i++)
        if(n % i == 0)
            return false;
    return true;
}

int Insert(int n , int m){
    int h;

    h = n % m;
    // 有空位的话插入即可
    if(hashTable[h] == 0){
        hashTable[h] = 1;
        return h;
    }
    else{
        int temp = h;
        // 平方探测，最多探测m次
        for(int i = 1;i &lt; m;i++){
            h = (temp + i * i) % m;
            if(hashTable[h] == 0){
                hashTable[h] = 1;
                return h;
            }
        }
        return -1;
    }
}
int main(){
    int m , n;
    int i;
    int num;
    int index;
    vector&lt;int&gt; v;

    cin &gt;&gt; m &gt;&gt; n;
    while(!IsPrime(m))
        m++;
    for(i = 0;i &lt; n;i++){
        cin &gt;&gt; num;
        // 插入哈希表
        index = Insert(num , m);
        v.push_back(index);
    }
    // 输出
    for(vector&lt;int&gt;::iterator it = v.begin();it != v.end();it++){
        if(it != v.begin())
            cout &lt;&lt; " ";
        if(*it == -1)
            cout &lt;&lt; "-";
        else
            cout &lt;&lt; *it;
    }
    cout &lt;&lt; endl;
    return 0;
}

PAT甲级1077 Kuchiguse(C++)

发表于 2017-02-25 | 分类于 PAT甲级

PAT甲级 Kuchiguse，原题地址：https://www.patest.cn/contests/pat-a-practise/1077
解题分析：
先将第一个字符串储存在res向量中，然后从第二个字符串开始比较，比较时从尾开始，一旦发现有不一样的字符，更新res向量，继续下一个字符串比较，如果没有发现不一样的字符，但非res向量的字符串更短，同样需要更新res向量。res向量为空时输出nai，否则输出res向量。

#include&lt;iostream&gt;
#include&lt;vector&gt;
#include&lt;algorithm&gt;
using namespace std;

int main(){
    vector&lt; vector&lt;char&gt; &gt; v;
    vector&lt;char&gt; temp;
    vector&lt;char&gt; res(300);
    vector&lt; vector&lt;char&gt; &gt;::iterator it;
    char ch;
    int n;
    int i , j , k;
    int size1 , size2;

    cin &gt;&gt; n;
    // 输入
    getchar();
    for(i = 0;i &lt; n;i++){
        while((ch = getchar()) != '\n'){
            temp.push_back(ch);
        }
        v.push_back(temp);
        temp.clear();
    }
    // 将第一个字符串拷贝到res中
    temp = *v.begin();
    copy(temp.begin() , temp.end() , res.begin());
    // 遍历其他字符串
    for(it = v.begin() + 1;it != v.end();it++){
        temp = *it;
        // 计算两个字符串的长度
        size1 = temp.size();
        for(i = 0,size2 = 0;res[i] != '\0';i++)
            size2++;
        // 从尾开始比较
        for(j = size1 - 1 , k = size2 - 1;j &gt;= 0 &amp;&amp; k &gt;= 0;j--,k--){
            // 如果有字符不相等
            if(temp[j] != res[k]){
                // 如果第一个尾字符就不相等，说明没有共同的后缀，输出nai
                if(j == size1 - 1){
                    printf("nai\n");
                    return 0;
                }
                // 否则更新res
                else{
                    res.clear();
                    res.resize(300);
                    copy(temp.begin() + j + 1 , temp.end() , res.begin());
                }
                break;
            }
            // j等于0说明非res字符串更短，更新res
            if(j == 0){
                res.clear();
                res.resize(300);
                copy(temp.begin() , temp.end() , res.begin());
            }
        }
    }
    // 输出
    for(i = 0;i &lt; res.size();i++)
        cout &lt;&lt; res[i];
    cout &lt;&lt; endl;
}

STL入门

发表于 2017-02-25 | 分类于 C++

最近在学习STL，看了网上的一些资料后稍微弄清楚了一些，这里主要记录STL容器的使用、迭代器的概念和使用以及一些通用的算法。

1、迭代器

迭代器用来遍历容器中的元素，顺序容器和关联容器都有各自对应的迭代器。

迭代器定义的格式为container::iterator variable，比如vector::iterator it;

迭代器的基本操作是和++，表示获取迭代器所指的内容，++表示迭代器向前移动一个元素。常用的迭代器有begin()和end()。

迭代器共有五种：

  • Input Iterator:支持单步向前迭代，但只能读取迭代器内容而不能修改
  • Output Iterator:支持单步向前迭代，但只能写不能读
  • Forward Iterator:支持单步向前迭代，拥有Input的全部特性和Output的部分特性
  • Bidirectional Iterator:在Forward的基础上支持向后迭代
  • Random Access Iterator:随机访问迭代器，不仅支持单步迭代，同时还可以进行算术计算

所有的迭代器都可以进行不等的判断，但只有随机访问迭代器才能比较迭代器的大小(>,>=,<,<=)，同时也只有随机访问迭代器支持算术计算，其他迭代器只能单步向前或向后迭代。

容器和相应的迭代器

  • vector和dequeue支持随机访问迭代器
  • list、set、map支持双向迭代器
  • stack和queue不支持迭代器

2、容器

容器用来储存数据。

(一)vector

(1)定义和初始化

  • vector<type> vec1; //定义一个vector容器vec1，容器中的元素类型为type;
  • vector<type> vec2(size); //定义一个初始大小为size的vector容器;
  • vector<type> vec3(size,value); //定义一个初始大小为size，每一个元素值为value的vector容器;
  • vector<type> vec4(another vector); //定义一个vector容器，它是另一个vector容器的副本;
  • vector<type> vec5(first,last); //根据两个迭代器之间的元素常见vector;

(2)基本操作

  • v.empty():检查v是否为空，是返回true，否则返回false
  • v.size():返回v的元素个数
  • v.push_back(t):在v的后面添加一个元素，v的长度加一
  • v.pop_back():删除v最后一个元素
  • v.resize(n):重新设置v的大小，n比现有元素数量大，则扩展v，不够的添0，否则截去多余的部分
  • v.erase(it):删除迭代器it所指的元素，返回被删除元素后一个元素的迭代器
  • v.erase(first,last):删除迭代器first和last之间的元素，返回被删段后面的元素的迭代器
  • v.clear():清除所有元素
  • v.insert(it,x):在迭代器it所指元素前插入x，返回新元素的迭代器
  • v.reverse():翻转元素顺序
  • v.swap(v1):和另一个vector交换，两个容器的元素类型应一致

(二)list

(1)定义和初始化

list的定义和初始化与vector类似。

(2)基本操作

  • l.front():返回链表第一个节点的值
  • l.back():返回链表最后一个节点的值
  • l.push_back(t):向链表尾部添加一个节点
  • l.push_front(t):向链表头部添加一个节点
  • l.pop_back():删除最后一个节点
  • l.pop_front():删除第一个节点
  • l.remove(val):删除所有值为val的节点
  • l.sort(prev):根据给定的谓词对链表进行排序，默认为升序

(三)dequeue

(1)定义和初始化

dequeue定义和初始化和vector类似。

(2)基本操作

  • d[]:支持下标运算
  • d.front():返回队列第一个元素的值
  • d.back():返回队列最后一个元素的值
  • d.push_front(t):向队列的头部插入一个新元素
  • d.push_back(t):向队列的尾部插入一个新元素
  • d.pop_front():删除队列的第一个元素
  • d.pop_back():删除队列的最后一个元素

(四)stack

(1)定义和初始化

stack<type> variable;

(2)基本操作

  • s.empty():检查栈是否为空，是返回true，否则返回false
  • s.size():返回栈中元素的个数
  • s.push(t):将元素压入栈顶
  • s.pop():将栈顶元素删除，但不返回其值
  • s.top():返回栈顶元素的值，但不删除

(五)queue

(1)定义和初始化

queue<type> variable;

(2)基本操作

  • q.empty():判断队列是否为空
  • q.size():返回队列中的元素个数
  • q.push():向队尾添加元素
  • q.pop():删除队首的元素，但不返回其值
  • q.front():返回队首元素的值
  • q.back():返回队尾元素的值

(六)map

(1)定义和初始化

  • map<key,value> m1; //定义一个空的map容器
  • map<key,value> m2(m); //m2是m的副本
  • map<key,value> m3(first,last); //根据两个迭代器之间的元素构建map
  • map<key,value,comp> m4; //根据comp决定key的存放顺序，默认使用<号比较，key必须可以进行比较

(2)基本操作

  • m[key]:下标访问，如果key不存在，则会建立一个key
  • m.cout(k):返回key为k的元素的个数，由于map的key不重复，因此返回值为0和1，用来检查某个key是否存在
  • m.find(k):返回key为k的元素的迭代器，如果不存在返回m.end()
  • m.erase(k):删除key为k的元素，返回被删除元素的个数
  • m.erase(it):删除迭代器所指的元素，迭代器不能为m.end()
  • m.insert(e):e为pair类型，如果e.first在m中不存在，则插入，返回一个pair类型的值，其第一个元素是指向插入元素的map迭代器，第二个元素为bool类型，表示是否插入

(七)set

(1)定义和初始化

常规定义方法。

(2)基本操作

  • s.insert(key):如果key不存在，插入key，否则不做任何操作，返回类型的值为pair类型，first表示插入的位置，second表示是否插入成功
  • s.insert(first,last):将两个迭代器之间的元素插入，返回void
  • s.find(key):查找key，找到则返回对于的迭代器，否则返回s.end()
  • s.erase(key):删除key
  • s.erase(it):删除迭代器所指元素

PAT甲级1076 Forwards on Weibo(C++)

发表于 2017-02-24 | 分类于 PAT甲级

PAT甲级1076 Forwards on Weibo，原题地址：https://www.patest.cn/contests/pat-a-practise/1076
解题分析：
BFS。整个网络其实就是一个有向图，一个节点可以指向一些节点，这些节点就是关注他的人。用二维数组储存有向图，然后根据查询的值，宽度有限搜索直到搜索的层次等于规定的层次或者没有节点可以搜索为止。需要注意的是，将节点加入到队列中时，就需要将该节点设为已被访问，否则可能出现队列中有重复的节点。

#include&lt;iostream&gt;
#include&lt;queue&gt;
using namespace std;
#define Max 1100

int G[Max][Max];
int visited[Max];
queue&lt;int&gt; Q;
int n , l;

int main(){
    int i , j;
    int k;
    int m;
    int u;
    int cnt;
    int level;
    int temp;
    int forward;

    cin &gt;&gt; n &gt;&gt; l;
    for(i = 1;i &lt;= n;i++){
        cin &gt;&gt; m;
        // 储存图
        for(j = 0;j &lt; m;j++){
            cin &gt;&gt; u;
            G[u][i] = 1;
        }
    }
    cin &gt;&gt; k;
    // 查询
    for(i = 0;i &lt; k;i++){
        cin &gt;&gt; u;
        // 初始化
        level = 0;
        forward = 0;
        // 全部节点设为未访问
        for(j = 1;j &lt;= n;j++)
            visited[j] = 0;
        // 将根节点加入队列
        Q.push(u);
        visited[u] = 1;
        // 搜索层次不能超过l
        while(level &lt; l){
            // 如果没有节点可以搜索则退出循环
            if(Q.empty())
                break;
            cnt = Q.size();
            // 遍历一层内所有节点
            while(cnt &gt; 0){
                temp = Q.front();
                Q.pop();
                cnt--;
                // 将子节点加入队列
                for(j = 1;j &lt;= n;j++){
                    if(G[temp][j] == 1 &amp;&amp; visited[j] == 0){
                        visited[j] = 1;
                        Q.push(j);
                        forward++;
                    }
                }
            }
            // 一层结束level加一
            level++;
        }
        // 清空队列
        while(!Q.empty())
            Q.pop();
        cout &lt;&lt; forward &lt;&lt; endl;
    }
}

PAT甲级1075 PAT Judge(C++)

发表于 2017-02-24 | 分类于 PAT甲级

PAT甲级1075 PAT Judge，原题地址：https://www.patest.cn/contests/pat-a-practise/1075
解题分析：
排序题。对于每一个用户，我们需要纪录他的id、每一题的得分、每一题的提交次数、总分以及通过的题数。初始时将总分设为－1，每一题得分设为－1，如果没有提交过任何题目或者没有一道题通过编译，那么总分始终是－1，排序后直接舍弃即可。排序时先根据总分，总分相同则按通过的题数，如果也相同再按id。输出时需要注意，如果有用户提交过题目并且至少有一题通过了编译，则该用户需要被输出，如果其还提交了其他题目，但没有通过编译，输出0，否则输出相应的分数。

#include&lt;iostream&gt;
#include&lt;cstdio&gt;
#include&lt;algorithm&gt;
using namespace std;

// 储存用户信息
struct record{
    int id;
    int score[6];
    int submit[6];
    int total;
    int perfect;
};

struct record R[11000];
int problem[6];

// 排序
bool cmp(struct record a , struct record b){
    // 先根据总分
    if(a.total &gt; b.total)
        return true;
    else if(a.total &lt; b.total)
        return false;
    // 再根据通过的题数
    else{
        if(a.perfect &gt; b.perfect)
            return true;
        else if(a.perfect &lt; b.perfect)
            return false;
        // 最后根据id
        else
            return a.id &lt; b.id;
    }
}

int main(){
    int n , k;
    int uid , pid , score;
    long m;
    long i , j;
    long start , end;
    int rank = 1;

    cin &gt;&gt; n &gt;&gt; k &gt;&gt; m;
    for(i = 1;i &lt;= k;i++)
        cin &gt;&gt; problem[i];
    // 舒适化
    for(i = 1;i &lt;= n;i++){
        R[i].id = i;
        R[i].total = -1;
        R[i].perfect = 0;
        for(j = 1;j &lt;= k;j++){
            R[i].score[j] = -1;
            R[i].submit[j] = 0;
        }
    }
    // 输入
    for(i = 0;i &lt; m;i++){
        cin &gt;&gt; uid &gt;&gt; pid &gt;&gt; score;
        // 如果分数大于原来的分数
        if(score &gt; R[uid].score[pid]){
            // 如果总分为－1
            if(R[uid].total == -1)
                R[uid].total = 0;
            // 如果之前的分数是－1，则总分直接加上现在的得分即可
            if(R[uid].score[pid] == -1){
                R[uid].total += score;
            }
            // 否则需要加上二者之差
            else{
                R[uid].total += score - R[uid].score[pid];
            }
            // 纪录得分
            R[uid].score[pid] = score;
            // 如果通过该题
            if(score == problem[pid])
                R[uid].perfect++;
        }
        // 该题提交次数加一
        R[uid].submit[pid]++;
    }
    // 排序
    sort(R + 1 , R + n + 1 , cmp);
    // 输出
    for(start = 1;start &lt;= n &amp;&amp; R[start].total &gt;= 0;start = end + 1){
        end = start + 1;
        while(end &lt;= n &amp;&amp; R[end].total == R[start].total)
            end++;
        end--;
        for(i = start;i &lt;= end;i++){
            printf("%d %05d %d" , rank , R[i].id , R[i].total);
            for(j = 1;j &lt;= k;j++){
                if(R[i].submit[j] == 0)
                    printf(" -");
                else if(R[i].score[j] == -1)
                    printf(" 0");
                else
                    printf(" %d" , R[i].score[j]);
            }
            printf("\n");
        }
        rank += end - start + 1;
    }
    return 0;
}
