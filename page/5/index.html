最大公约数和最小公倍数

发表于 2017-02-28 | 分类于 杂记 | 阅读次数

快速求得最大公约数的方法－辗转相除法

int gcd(int a , int b){
    int c;

    while(a % b != 0){
        c = a % b;
        a = b;
        b = c;
    }
    return b;
}

返回的b就是原来的a和b的最大公约数。

最小公倍数 = a * b / 最大公约数。

PAT甲级1090 Highest Price in Supply Chain(C++)

发表于 2017-02-28 | 分类于 PAT甲级 | 阅读次数

PAT甲级1090 Highest Price in Supply Chain，原题地址：https://www.patest.cn/contests/pat-a-practise/1090
解题分析：
题目其实很简单，用DFS遍历一遍算出最高的溢价率就可以了。但问题在于花费的时间，每次搜索到一个点，我们都需要找到这个点的子节点，如果我们每次都通过简单遍历来得到子节点，那么时间复杂度为O(N^2)，提交上去会超时。因此，我们需要换一种方法来查找，最简单的就是二分查找，使用二分查找能将时间复杂度降低到O(NlogN)，可以通过测试。

#include&lt;iostream&gt;
#include&lt;cstdio&gt;
#include&lt;vector&gt;
#include&lt;algorithm&gt;
using namespace std;

typedef long long ll;
struct node{
    ll v;
    ll p;
};

ll n;
double p , r;
vector&lt;struct node&gt; c;
double Max;
ll cnt;
ll t;

// 二分查找
ll BinarySearch(ll value){
    ll start = 0 , end = n - 1 , mid;

    while(start &lt;= end){
        mid = (start + end) / 2;
        if(c[mid].v == value)
            return mid;
        else if(c[mid].v &gt; value)
            end = mid - 1;
        else
            start = mid + 1;
    }
    return -1;
}

// 深度搜索
void DFS(ll root , double rate){
    ll i;
    ll index;
    ll start , end;
    bool flag = false;

    // 根节点的溢价率为0，因此不需要计算
    if(root != c[0].p)
        rate *= (1 + r / 100);
    // 找到子节点在c中的索引
    index = BinarySearch(root);
    // 没找到说明是零售商
    if(index &lt; 0){
        // 比较溢价率和最大溢价率，比它大就把它更新
        if(rate &gt; Max){
            Max = rate;
            cnt = 1;
        }
        // 跟它相等计数加一
        else if(rate == Max)
            cnt++;
        return;
    }
    // 获取所有字节点
    start = index - 1;
    end = index + 1;
    while(start &gt;= 0 &amp;&amp; c[start].v == root) start--;
    while(end &lt; n &amp;&amp; c[end].v == root) end++;
    // 继续搜索
    for(i = start + 1;i &lt; end;i++)
        DFS(c[i].p , rate);
}

bool cmp(struct node a , struct node b){
    return a.v &lt; b.v;
}

int main(){
    ll i;
    ll num;
    struct node R;

    cin &gt;&gt; n &gt;&gt; p &gt;&gt; r;
    for(i = 0;i &lt; n;i++){
        cin &gt;&gt; num;
        R.v = num;
        R.p = i;
        c.push_back(R);
    }
    // 排序，以便使用二分查找
    sort(c.begin() , c.end() , cmp);
    DFS(c[0].p , 1);
    printf("%.2lf %lld\n" , Max * p , cnt);
}

PAT甲级1089 Insert or Merge(C++)

发表于 2017-02-28 | 分类于 PAT甲级 | 阅读次数

PAT甲级1089 Insert or Merge，原题地址：https://www.patest.cn/contests/pat-a-practise/1089
解题分析：
插入排序和归并排序。插入排序很简单，遍历所有的元素，一个一个插入就行了，每插入一个元素检查是否和所给的序列相等，如果是则再执行一次排序后返回。归并排序有些复杂，这里不能用递归来写归并排序，因为如果用递归写归并的话，每次排序的元素个数是2、2、4、2、2、4、8，这跟题目的要求不符合，从题目中我们可以看到，它每次排序的元素个数是2、4、8，因此我们需要用归并排序的非递归方法来解题，这一点明确后题目就不难了。

#include&lt;iostream&gt;
#include&lt;vector&gt;
using namespace std;

vector&lt;int&gt; v1;
vector&lt;int&gt; v2;
vector&lt;int&gt; t;
int n;

// 归并两个排好序的序列
void merge(int start , int end , int len){
    vector&lt;int&gt; temp;
    int mid = start + len / 2 - 1;
    int i , j;

    // 比较两个序列中的元素大小，然后依次放入temp中
    for(i = start , j = mid + 1;i &lt;= mid &amp;&amp; j &lt;= end;){
        if(t[i] &lt; t[j]){
            temp.push_back(t[i]);
            i++;
        }
        else{
            temp.push_back(t[j]);
            j++;
        }
    }

    // 处理剩下的元素
    while(i &lt;= mid){
        temp.push_back(t[i]);
        i++;
    }
    while(j &lt;= end){
        temp.push_back(t[j]);
        j++;
    }

    // 修改t
    for(i = 0;i &lt; temp.size();i++){
        t[start++] = temp[i];
    }
}

// 判断是否和给定序列相等
bool IsEqual(){
    int i;

    for(i = 0;i &lt; n;i++){
        if(t[i] != v2[i])
            return false;
    }
    return true;
}

// 判断是否是插入排序
bool IsInsertion(){
    int i , j , k;
    int temp;
    int flag = 0;

    t.clear();
    for(i = 0;i &lt; n;i++)
        t.push_back(v1[i]);
    for(i = 1;i &lt; n;i++){
        // 找到元素插入的位置
        for(j = 0;j &lt; i;j++){
            if(t[j] &gt; t[i])
                break;
        }
        temp = t[i];
        // 移动元素
        for(k = i - 1;k &gt;= j;k--)
            t[k + 1] = t[k];
        // 插入
        t[j] = temp;
        if(flag)
            return true;
        if(IsEqual())
            flag = 1;
    }
    return false;
}

// 判断是否是归并排序
bool IsMerge(){
    int i;
    int start , end;
    int flag = 0;

    t.clear();
    for(i = 0;i &lt; n;i++)
        t.push_back(v1[i]);
    // 每次排序的元素个数是2、4、8
    for(i = 2;i &lt;= n;i *= 2){
        // 划分区间
        for(start = 0;start &lt; n;start = end + 1){
            end = start + i - 1;
            if(end &gt;= n)
                end = n - 1;
            // 排序
            merge(start , end , i);
        }
        if(flag)
            return true;
        if(IsEqual())
            flag = 1;
    }
    return false;
}

int main(){
    int num;
    int i;

    // 输入
    cin &gt;&gt; n;
    for(i = 0;i &lt; n;i++){
        cin &gt;&gt; num;
        v1.push_back(num);
    }
    for(i = 0;i &lt; n;i++){
        cin &gt;&gt; num;
        v2.push_back(num);
    }
    // 判断
    if(IsInsertion()){
        cout &lt;&lt; "Insertion Sort" &lt;&lt; endl;
        for(i = 0;i &lt; n;i++){
            if(i != n - 1)
                cout &lt;&lt; t[i] &lt;&lt; " ";
            else
                cout &lt;&lt; t[i];
        }
    }
    else if(IsMerge()){
        cout &lt;&lt; "Merge Sort" &lt;&lt; endl;
        for(i = 0;i &lt; n;i++){
            if(i != n - 1)
                cout &lt;&lt; t[i] &lt;&lt; " ";
            else
                cout &lt;&lt; t[i];
        }
    }
}

#include&lt;cstdio&gt;

long long gcd(long long a , long long b){
    long long c;

    a = a &gt; 0 ? a : -a;
    b = b &gt; 0 ? b : -b;
    while(a % b != 0){
        c = a % b;
        a = b;
        b = c;
    }
    return b;
}

long long GetCommon(long long a , long long b){
    long long t;

    t = gcd(a , b);
    return a * b / t;
}

void Simplify(long long &amp;a , long long &amp;b){
    int t;

    if(b == 0)
        return;
    t = gcd(a , b);
    a /= t;
    b /= t;
}

void Output(long long a , long long b){
    if(a &gt;= 0 &amp;&amp; b &gt;= 0){
        if(a % b == 0)
            printf("%lld" , a / b);
        else if(a &lt; b)
            printf("%lld/%lld" , a , b);
        else
            printf("%lld %lld/%lld" , a / b , a % b , b);
    }
    else{
        printf("(-");
        if(a &lt; 0)
            a = -a;
        if(b &lt; 0)
            b = -b;
        if(a % b == 0)
            printf("%lld" , a / b);
        else if(a &lt; b)
            printf("%lld/%lld" , a , b);
        else
            printf("%lld %lld/%lld" , a / b , a % b , b);
        printf(")");
    }
}

int main(){

    Simplify(c1 , p1);
    Simplify(c2 , p2);

    pres = GetCommon(p1 , p2);
    cres = pres / p1 * c1 + pres / p2 * c2;
    Simplify(cres , pres);
    Output(c1 , p1);
    printf(" + ");
    Output(c2 , p2);
    printf(" = ");
    Output(cres , pres);
    printf("\n");

    pres = GetCommon(p1 , p2);
    cres = pres / p1 * c1 - pres / p2 * c2;
    Simplify(cres , pres);
    Output(c1 , p1);
    printf(" - ");
    Output(c2 , p2);
    printf(" = ");
    Output(cres , pres);

PAT甲级1087 All Roads Lead to Rome(C++)

发表于 2017-02-27 | 分类于 PAT甲级 | 阅读次数

PAT甲级1087 All Roads Lead to Rome，原题地址：https://www.patest.cn/contests/pat-a-practise/1087
解题分析：
Dijkstra加DFS。先用Dijkstra求出Rome的最短路径，然后从起点出发DFS，搜索时保存走过的路长，如果比Rome的最短路径长直接返回，以此减少程序的运行时间。搜索到Rome时，需要和之前的最优路径，如果happy值比它高，则最优路径为当前路径，如果相等，再比较平均happy值，如果比它高，则最优路径也为当前路径。用容器储存路径上的点，最后输出即可。

#include&lt;iostream&gt;
#include&lt;string&gt;
#include&lt;vector&gt;
#include&lt;algorithm&gt;
using namespace std;
#define Max 210
#define INFINITY 100000

int G[Max][Max];
int hp[Max];
int Distance[Max];
int Visited[Max];
int currentDistance;
int currentFun , fun , avfun;
int cnt , num;
int t;
vector&lt;string&gt; v1;
vector&lt;string&gt; v2;   
int n , k;

// 最短路径算法，太常用了就不解释了
void Dijkstra(){
    int i;
    int index , min;

    while(1){
        min = INFINITY;
        for(i = 0;i &lt; n;i++){
            if(Visited[i] == 0 &amp;&amp; Distance[i] &lt; min){
                min = Distance[i];
                index = i;
            }
        }
        if(min == INFINITY)
            break;
        Visited[index] = 1;
        for(i = 0;i &lt; n;i++){
            if(G[index][i] &gt; 0 &amp;&amp; Visited[i] == 0 &amp;&amp; G[index][i] + Distance[index] &lt; Distance[i])
                Distance[i] = G[index][i] + Distance[index];
        }
    }
}

// DFS
bool DFS(int start){
    bool flag = false;
    int i;

    // 剪枝
    if(currentDistance &gt; Distance[t])
        return false;

    // 更新一些全局变量
    num++;
    currentFun += hp[start];
    Visited[start] = 1;
    // 如果是Rome点
    if(start == t){
        // 路径条数加一
        cnt++;
        // 如果happy比它大
        if(currentFun &gt; fun){
            fun = currentFun;
            avfun = fun / (num - 1);
            v2.clear();
            v2.push_back(v1[start]);
            flag = true;
        }
        // 如果happy值相等，但平均happy比它大
        else if(currentFun == fun &amp;&amp; currentFun / (num - 1) &gt; avfun){
            avfun = currentFun / (num - 1);
            v2.clear();
            v2.push_back(v1[start]);
            flag = true;
        }
    }
    // 不是Rome点
    else{
        for(i = 0;i &lt; n;i++){
            // 遍历其他点
            if(G[start][i] &gt; 0 &amp;&amp; Visited[i] == 0){
                currentDistance += G[start][i];
                if(DFS(i))
                    flag = true;
                currentDistance -= G[start][i];
            }
        }
        // 如果当前点在路径上
        if(flag)
            v2.push_back(v1[start]);
    }

    // 重新更新全局变量
    num--;
    currentFun -= hp[start];
    Visited[start] = 0;
    if(flag)
        return true;
    else
        return false;
}
int main(){
    string s , a , b;
    int i;
    int temp;
    int t1 , t2;

    cin &gt;&gt; n &gt;&gt; k &gt;&gt; s;
    v1.push_back(s);
    for(i = 1;i &lt;= n - 1;i++){
        cin &gt;&gt; s &gt;&gt; temp;
        // v1容器保存字符串
        v1.push_back(s);
        // hp数组保存happy值
        hp[i] = temp;
    }
    for(i = 0;i &lt; k;i++){
        cin &gt;&gt; a &gt;&gt; b &gt;&gt; temp;
        // 求出对应字符串的序列号
        t1 = find(v1.begin() , v1.end() , a) - v1.begin();
        t2 = find(v1.begin() , v1.end() , b) - v1.begin();
        // 储存图信息
        G[t1][t2] = temp;
        G[t2][t1] = temp;
    }
    // 初始化
    for(i = 0;i &lt; n;i++){
        Distance[i] = INFINITY;
        Visited[i] = 0;
    }
    // 求最短路径
    Distance[0] = 0;
    Dijkstra();
    // Rome的索引
    t = find(v1.begin() , v1.end() , "ROM") - v1.begin();
    for(i = 0;i &lt; n;i++)
        Visited[i] = 0;
    // DFS
    DFS(0);
    // 输出
    cout &lt;&lt; cnt &lt;&lt; " " &lt;&lt; Distance[t] &lt;&lt; " " &lt;&lt; fun &lt;&lt; " " &lt;&lt;  avfun &lt;&lt; endl;
    for(i = v2.size() - 1;i &gt;= 0;i--){
        cout &lt;&lt; v2[i];
        if(i != 0)
            cout &lt;&lt; "-&gt;";
    }
}

PAT甲级1086 Tree Traversals Again(C++)

发表于 2017-02-27 | 分类于 PAT甲级 | 阅读次数

PAT甲级1086 Tree Traversals Again，原题地址：https://www.patest.cn/contests/pat-a-practise/1086
解题分析：
树的遍历。可以用栈来模拟中序遍历，也可以通过栈的push和pop顺序来还原树。用容器v作为栈，push的时候将输入的值放入v中，然后将值插入到树中，重点在于，插入的时候要判断当前节点是否在栈中，如果在，则插在它的左子树，如果不在，则插在右子树。pop的时候将值从v山中删除即可。然后简单的递归后序遍历树输出即可。

#include&lt;iostream&gt;
#include&lt;vector&gt;
#include&lt;string&gt;
#include&lt;algorithm&gt;
using namespace std;

struct node{
    int index;
    struct node *left;
    struct node *right;
};

vector&lt;int&gt; v;
int flag = 1;

// 插入
struct node *Insert(struct node *root , int index){
    struct node *ptr;

    // 根节点为空，则新建一个节点后返回
    if(root == NULL){
        ptr = new struct node;
        ptr-&gt;index = index;
        ptr-&gt;left = ptr-&gt;right = NULL;
        root = ptr;
    }
    // 如果根节点不为空
    else{
        // 在v中查找根节点的值是否存在
        if(find(v.begin() , v.end() , root-&gt;index) != v.end()){
            // 如果存在，插在左子树
            root-&gt;left = Insert(root-&gt;left , index);
        }
        else{
            // 如果不存在，插在右子树
            root-&gt;right = Insert(root-&gt;right , index);
        }
    }
    return root;
}

// 后序遍历
void PostOrder(struct node *root){
    if(root == NULL)
        return;
    PostOrder(root-&gt;left);
    PostOrder(root-&gt;right);
    if(flag){
        flag = 0;
        cout &lt;&lt; root-&gt;index;
    }
    else
        cout &lt;&lt; " " &lt;&lt; root-&gt;index;
}

int main(){
    struct node *root = NULL;
    int n;
    int i;
    int num;
    string s;

    cin &gt;&gt; n;
    for(i = 0;i &lt; 2 * n;i++){
        cin &gt;&gt; s;
        // push
        if(s[1] == 'u'){
            cin &gt;&gt; num;
            v.push_back(num);
            root = Insert(root , num);
        }
        // pop
        else{
            // 删去最后一个元素
            v.resize(v.size() - 1);
        }
    }
    // 后序遍历
    PostOrder(root);
}

PAT甲级1085 Perfect Sequence(C++)

发表于 2017-02-27 | 分类于 PAT甲级 | 阅读次数

PAT甲级1085 Perfect Sequence，原题地址：https://www.patest.cn/contests/pat-a-practise/1085
解题分析：
用两个指针循环，可以得到所有符合条件的序列，选择其中最长的即可。

#include&lt;iostream&gt;
#include&lt;algorithm&gt;
using namespace std;
#define Max 100000

int main(){
    long n , p;
    long arr[Max];
    long i , j;
    long ans = 0;

    cin &gt;&gt; n &gt;&gt; p;
    for(i = 0;i &lt; n;i++)
        cin &gt;&gt; arr[i];
    sort(arr , arr + n);
    for(i = 0;i &lt; n;i++){
        while(j &lt; n &amp;&amp; arr[j] &lt;= arr[i] * p) j++;
        ans = ans &gt; j - i ? ans : j - i;
    }
    cout &lt;&lt; ans &lt;&lt; endl;
}

PAT甲级1084 Broken Keyboard(C++)

发表于 2017-02-27 | 分类于 PAT甲级 | 阅读次数

PAT甲级1084 Broken Keyboard，原题地址：https://www.patest.cn/contests/pat-a-practise/1084
解题分析：
遍历第二个字符串，其中每一个字符都在第一个字符串中搜索，这样就可以获得一个区间，对区间内的字符，首先在容器v中搜索，如果没有就将它放入，如果有则跳过。第二个字符串遍历完之后，如果第一个字符串还有剩余，执行区间内字符相同的操作。

#include&lt;iostream&gt;
#include&lt;string&gt;
#include&lt;vector&gt;
#include&lt;algorithm&gt;
using namespace std;

int main(){
    string str1 , str2;
    vector&lt;char&gt; v;
    int start , end;
    int i, j;

    cin &gt;&gt; str1 &gt;&gt; str2;
    start = end = 0;
    // 遍历第二个字符串
    for(i = 0;i &lt; str2.length();i++){
        // 划分区间
        while(str1[end] != str2[i])
            end++;
        // 遍历区间内的字符
        for(j = start;j &lt; end;j++){
            // 小写变成大写
            if(str1[j] &gt;= 'a'&amp;&amp; str1[j] &lt;= 'z')
                str1[j] = str1[j] - 'a' + 'A';
            // 在v查找
            if(find(v.begin(), v.end() , str1[j]) == v.end())
                v.push_back(str1[j]);
        }
        start = end + 1;
        end = start;
    }
    // 第一个字符串还有剩余
    while(start &lt;str1.length()){
        if(str1[start] &gt;= 'a' &amp;&amp; str1[start] &lt;= 'z')
            str1[start] = str1[start] - 'a' + 'A';
        if(find(v.begin(), v.end() , str1[start]) == v.end())
            v.push_back(str1[start]);
        start++;
    }

Strict Weak Ordering in C++

发表于 2017-02-26 | 分类于 C++ | 阅读次数

最近刷PAT的时候，有两道题因为段错误死活过不去，调试的时候发现是STL中的sort出现了问题，第一次的时候不知道怎么修改就放弃重写了，第二次又遇到我就很生气了，于是就在网上找了一些资料，最后将问题定位到了strict weak ordering(严格弱排序)上，在这里简单的记录一下。

首先，我们来看一下维基百科上关于严格弱排序的定义：

A strict weak ordering is a binary relation < on a set S that is a strict partial order (a transitive relation that is irreflexive, or equivalently,that is asymmetric) in which the relation “neither a < b nor b < a” is transitive.Therefore, a strict weak ordering has the following properties:
1、For all x in S, it is not the case that x < x (irreflexivity).
2、For all x, y in S, if x < y then it is not the case that y < x (asymmetry).
3、For all x, y, z in S, if x < y and y < z then x < z (transitivity).
4、For all x, y, z in S, if x is incomparable with y (neither x < y nor y < x hold), and y is incomparable with z, then x is incomparable with z (transitivity of incomparability).
The “incomparability relation” is an equivalence relation, and its equivalence classes partition the elements of S, and are totally ordered by <.

严格弱排序就是定义在集合S上的一种二元关系(relation)，这种二元关系就是“<”。这里的小于号不是普通的小于号，它代表的是一种关系，而不是简单的比大小。严格弱排序有四种性质，我们就看前两种吧。第一条，集合中的任何元素x，它跟自身都不满足“<”这种关系。第二条，集合中的两个元素x，y，如果x<y，那么y<x一定不成立。虽然看起来像是废话，但这就是数学的严谨性。

最重要的一点是，如果x<y不成立并且y<x也不成立，写成符号表达式就是!(x<y)&&!(y<x) = true，那么x和y就是不可比的(imcomparable)，根据定义，它们之间就是等价关系。理解了这一点，我们才能理解为什么sort会崩溃。

接下来我们看看STL官方文档对严格弱排序的描述：

A Strict Weak Ordering is a Binary Predicate that compares two objects, returning true if the first precedes the second. This predicate must satisfy the standard mathematical definition of a strict weak ordering.

根据这种描述，严格弱排序就是一个比较两个对象的二元谓词，或者说二元函数，这个函数返回true如果第一个对象先于第二个对象，同时它还要满足严格弱排序的数学定义。因此，如果函数返回true，即如果第一个对象(x)先于第二个对象(y)，我们可以认为x<y关系成立。

有了上面这些描述，我们就来具体看看sort函数。C++中的sort函数可以接受两个随机迭代器和一个比较函数，但规定了比较函数必须是严格弱排序的。sort函数在对序列排序时，必定会判断两个元素的相对位置应该是怎样的，那么sort是如何判断的呢？下面是我的一些理解，由于我还是初学C++，对sort的实现机制不是很了解，因此更多的是我的一些推测和猜想，但也能解释为什么sort函数会崩溃。

假设有两个对象x，y和一个比较函数f(a,b)。为了确定x和y的相对位置，首先计算f(x,y)，如果返回true，说明x<y，如果返回false，说明!(x<y)，注意这里不是y<x。然后计算f(y,x)，如果返回true，说明y<x，如果返回false，说明!(y<x)。x和y之间存在三种关系，x应该先于y，y应该先于x，x和y等价。对于前两种关系，只要一个返回true另一个返回false，sort就能正确地排出它们的顺序，而对于x和y等价，sort就需要两个函数都返回false。而我之前sort函数出现段错误的原因，就是我将比较函数定义成了在x和y等价时返回true。当用这种比较函数比较等价的x和y时，sort函数发现两个函数返回都是true，而这是一种没有定义的情况，可能就会触发某种机制，从而产生段错误。避免这种错误的方法就是在定义比较函数时，一定要确保两个对象等价时返回false。

PAT甲级1083 List Grades(C++)

发表于 2017-02-26 | 分类于 PAT甲级 | 阅读次数

PAT甲级1083 List Grades，原题地址：https://www.patest.cn/contests/pat-a-practise/1083
解题分析：
把不符合条件的剔除，剩下的简单的排下序就好了。

#include&lt;iostream&gt;
#include&lt;vector&gt;
#include&lt;string&gt;
#include&lt;algorithm&gt;
using namespace std;

struct record{
    string name;
    string id;
    int grade;
};

vector&lt;struct record&gt; v;
vector&lt;struct record&gt; ans;
long n;
int b1 , b2;

bool cmp(struct record a , struct record b){
    return a.grade &gt; b.grade;
}
int main(){
    struct record R;
    string name , id;
    int grade;
    int i;
    int flag = 0;

    cin &gt;&gt; n;
    for(i = 0;i &lt; n;i++){
        cin &gt;&gt; name &gt;&gt; id &gt;&gt; grade;
        R.name = name;
        R.id = id;
        R.grade = grade;
        v.push_back(R);
    }
    cin &gt;&gt; b1 &gt;&gt; b2;
    for(i = 0;i &lt; n;i++){
        if(v[i].grade &gt;= b1 &amp;&amp; v[i].grade &lt;= b2)
            ans.push_back(v[i]);
    }
    sort(ans.begin() , ans.end() , cmp);
    for(i = 0;i &lt; ans.size();i++){
        if(ans[i].grade &gt;= b1 &amp;&amp; ans[i].grade &lt;= b2){
            flag = 1;
            cout &lt;&lt; ans[i].name &lt;&lt; " " &lt;&lt; ans[i].id &lt;&lt; endl;
        }
        else{
            break;
        }
    }
    if(!flag)
        cout &lt;&lt; "NONE" &lt;&lt; endl;
    return 0;
}
