水题，不解释。

#include&lt;iostream&gt;
#include&lt;cstdio&gt;
using namespace std;

int arr[10000];
int flag[10000];

bool IsPrime(int n){
    int i;

    if(n == 1)
        return false;
    for(i = 2;i &lt; n;i++){
        if(n % i == 0)
            return false;
    }
    return true;
}

int main(){
    int n , k;
    int i;
    int id;
    int cnt = 1;

    cin &gt;&gt; n;
    for(i = 0;i &lt; n;i++){
        cin &gt;&gt; id;
        arr[id] = cnt;
        cnt++;
    }

    cin &gt;&gt; k;
    for(i = 0;i &lt; k;i++){
        cin &gt;&gt; id;
        printf("%04d: " , id);
        if(flag[id] == 1){
            cout &lt;&lt; "Checked" &lt;&lt; endl;
        }
        else if(arr[id] == 0){
            cout &lt;&lt; "Are you kidding?" &lt;&lt; endl;
        }
        else{
            flag[id] = 1;
            if(arr[id] == 1){
                cout &lt;&lt; "Mystery Award" &lt;&lt; endl;
            }
            else if(IsPrime(arr[id])){
                cout &lt;&lt; "Minion" &lt;&lt; endl;
            }
            else{
                cout &lt;&lt; "Chocolate" &lt;&lt; endl;
            }
        }
    }
}
