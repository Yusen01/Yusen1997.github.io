#include&lt;iostream&gt;
#include&lt;vector&gt;
#include&lt;algorithm&gt;
using namespace std;

int GetId(int n){
    int sum = 0;

    while(n &gt; 0){
        sum += n % 10;
        n /= 10;
    }
    return sum;
}
int main(){
    int arr[100];
    int i;
    int n;
    int t;
    int cnt = 0;
    vector&lt;int&gt; v;

    for(i = 0;i &lt; 100;i++){
        arr[i] = 0;
    }
    cin &gt;&gt; n;
    for(i = 0;i &lt; n;i++){
        int num;
        cin &gt;&gt; num;
        t = GetId(num);
        arr[t]++;
    }
    for(i = 0;i &lt; 100;i++){
        if(arr[i] &gt; 0){
            cnt++;
            v.push_back(i);
        }
    }
    sort(v.begin() , v.end());
    cout &lt;&lt; cnt &lt;&lt; endl;
    for(i = 0;i &lt; v.size();i++){
        cout &lt;&lt; v[i];
        if(i != v.size() - 1)
            cout &lt;&lt; " ";
    }
    cout &lt;&lt; endl;
}
