PAT甲级1127 ZigZagging on a Tree(C++)

发表于 2017-03-13 | 分类于 PAT甲级

PAT甲级1127 ZigZagging on a Tree，原题地址：https://www.patest.cn/contests/pat-a-practise/1127
解题分析：
用后序遍历序列和中序遍历序列构建一个二叉树，然后层序遍历，这个zigzag输出也很简单，判断一下层数是奇数还是偶数就可以了。。。

#include&lt;iostream&gt;
#include&lt;vector&gt;
#include&lt;queue&gt;
using namespace std;

struct node{
    int key;
    struct node *left;
    struct node *right;
};

int preOrder[50];
int postOrder[50];
int n;
int Index;

// 构建二叉树
struct node *Construct(int start , int end){
    struct node *ptr;
    int i;

    if(start &gt; end)
        return NULL;
    ptr = new struct node;
    ptr-&gt;key = postOrder[Index];
    for(i = start;i &lt;= end;i++){
        if(preOrder[i] == postOrder[Index]) 
            break;
    }
    Index--;
    ptr-&gt;right = Construct(i + 1 , end);
    ptr-&gt;left = Construct(start , i - 1);
    return ptr;
}

// 层序遍历
void LevelOrder(struct node *root){
    int cnt = 1;
    int flag = 1;
    queue&lt;struct node *&gt; q;
    vector&lt;int&gt; v;

    q.push(root);
    while(!q.empty()){
        int t = q.size();
        v.clear();
        while(t &gt; 0){
            struct node *ptr = q.front();
            q.pop();
            if(ptr-&gt;left != NULL)
                q.push(ptr-&gt;left);
            if(ptr-&gt;right != NULL)
                q.push(ptr-&gt;right);
            t--;
            v.push_back(ptr-&gt;key);
        }
        if(cnt % 2 != 0){
            for(int i = v.size() - 1;i &gt;= 0;i--){
                if(flag){
                    flag = 0;
                }
                else{
                    cout &lt;&lt; " ";
                }
                cout &lt;&lt; v[i];
            }
        }
        else{
            for(int i = 0;i &lt; v.size();i++){
                if(flag){
                    flag = 0;
                }
                else{
                    cout &lt;&lt; " ";
                }
                cout &lt;&lt; v[i];
            }
        }
        cnt++;
    }
}
int main(){
    int i;
    struct node *root = NULL;

    cin &gt;&gt; n;
    for(i = 0;i &lt; n;i++){
        cin &gt;&gt; preOrder[i];
    }
    for(i = 0;i &lt; n;i++){
        cin &gt;&gt; postOrder[i];
    }
    Index = n - 1;
    root = Construct(0 , n - 1);
    LevelOrder(root);
}
