PAT甲级1119 Pre- and Post-order Traversals(C++)

发表于 2017-03-13 | 分类于 PAT甲级 | 阅读次数

PAT甲级1119 Pre- and Post-order Traversals，原题地址：https://www.patest.cn/contests/pat-a-practise/1119
解题分析：
用中序和后序遍历得到的序列构建一个二叉树，二叉树可能并不唯一，大概的描述比较抽象，具体解释就直接写在注释里了。

#include&lt;iostream&gt;
using namespace std;

struct node{
    int key;
    struct node *left;
    struct node *right;
};

int preOrder[100] , postOrder[100];
int n;
int flag = 1;//flag标记二叉树是否唯一

// 递归构建二叉树
struct node *Construct(int index , int start , int end){
    struct node *ptr;

    // 递归出口
    if(index &gt;= n || start &gt; end){
        return NULL;
    }
    // 构建新结点
    ptr = new struct node;
    // 值为前序序列中对应的值
    ptr-&gt;key = preOrder[index];
    // 如果只有一个结点
    if(start == end){
        ptr-&gt;left = ptr-&gt;right = NULL;
        return ptr;
    }
    // 判断当前结点的后一个结点是否和后序序列的倒数第二个值相等，如果
    // 相等说明当前结点是后面所有结点的根结点，但当前结点可以是它的父
    // 结点的左孩子或右孩子
    if(postOrder[end - 1] == preOrder[index + 1]){
        // flag变为0
        flag = 0;
        // 假设是左孩子
        ptr-&gt;left = Construct(index + 1 , start , end - 1);
        ptr-&gt;right = NULL;
    }
    // 如果当前结点的后一个结点和后序序列的第一个值相等，则当前结点是它的父结点的左孩子，
    // 其他后面的结点全部位于它的父结点的右子树
    else if(postOrder[start] == preOrder[index + 1]){
        struct node *temp;
        temp = new struct node;
        temp-&gt;key = preOrder[index + 1];
        temp-&gt;left = temp-&gt;right = NULL;
        ptr-&gt;left = temp;
        // 构建右子树
        ptr-&gt;right = Construct(index + 2 , start + 1 , end - 1);
    }
    // 当前结点的后一个结点的值位于后序序列的中间
    else{
        int i;
        // 找到值的位置
        for(i = start + 1;i &lt; end - 1;i++){
            if(postOrder[i] == preOrder[index + 1])
                break;
        }
        // 构建左子树
        ptr-&gt;left = Construct(index + 1 , start , i);
        // 构建右子树
        ptr-&gt;right = Construct(index + i - start + 2 , i + 1 , end - 1);
    }
    return ptr;
}

// 中序遍历
void Inorder(struct node *root){
    if(root == NULL)
        return;
    Inorder(root-&gt;left);
    if(flag){
        cout &lt;&lt; root-&gt;key;
        flag = 0;
    }
    else{
        cout &lt;&lt; " " &lt;&lt; root-&gt;key;
    }
    Inorder(root-&gt;right);
}
int main(){
    int i;
    struct node *root;

    cin &gt;&gt; n;
    for(i = 0;i &lt; n;i++){
        cin &gt;&gt; preOrder[i]; 
    }
    for(i = 0;i &lt; n;i++){
        cin &gt;&gt; postOrder[i];
    }
    int index = 0;
    root = Construct(index , 0 , n - 1);
    if(flag){
        cout &lt;&lt; "Yes" &lt;&lt; endl;
    }
    else{
        cout &lt;&lt; "No" &lt;&lt; endl;
    }
    flag = 1;
    Inorder(root);
    cout &lt;&lt; endl;
}
